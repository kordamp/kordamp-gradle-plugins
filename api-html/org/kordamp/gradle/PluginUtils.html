047 @CompileStatic
048 class PluginUtils {
049     static boolean isAndroidProject(Project project) {
050         androidPlugins().any { project.pluginManager.hasPlugin(it) }
051     }
052 
053     /**
054      * Returns the sourceSets associated with the project.
055      *
056      * Return type is {@code SourceSetContainer} for a Java/Groovy/Groovy project or
057      * {@code AndroidSourceSet} if the project is Android related.
058      */
059     @CompileDynamic
060     static resolveSourceSets(Project project) {
061         if (isAndroidProject(project)) {
062             return project.android.sourceSets
063         }
064         project.extensions.findByType(SourceSetContainer)
065     }
066 
067     @CompileDynamic
068     static resolveSourceSets(Collection<Project> projects) {
069         projects.collect { resolveSourceSets(it) }
070     }
071 
072     @CompileDynamic
073     static resolveMainSourceDirs(Project project) {
074         if (isAndroidProject(project)) {
075             return project.android.sourceSets.main.javaDirectories.flatten()
076         }
077         project.sourceSets.main.allSource.srcDirs.flatten()
078     }
079 
080     @CompileDynamic
081     static resolveSourceDirs(Project project) {
082         if (isAndroidProject(project)) {
083             return project.android.sourceSets.javaDirectories.flatten()
084         }
085         project.sourceSets.collect { it.allSource.srcDirs }.flatten()
086     }
087 
088     static resolveMainSourceDirs(Collection<Project> projects) {
089         projects.collect { resolveMainSourceDirs(it) }.flatten()
090     }
091 
092     static resolveSourceDirs(Collection<Project> projects) {
093         projects.collect { resolveSourceDirs(it) }.flatten()
094     }

114     static boolean checkFlag(String flag, boolean defaultValue) {
115         if (isBlank(System.getProperty(flag))) {
116             return defaultValue
117         }
118         return Boolean.getBoolean(flag)
119     }

121     static boolean isGradleCompatible(Project project, int majorVersion) {
122         Version version = Version.of(GradleVersion.current().baseVersion.version)
123         version.major >= majorVersion
124     }
125 
126     static boolean supportsApiConfiguration(Project project) {
127         project.pluginManager.hasPlugin('java-library')
128     }

130     static AdhocComponentWithVariants findJavaComponent(SoftwareComponentContainer components) {
131         SoftwareComponent component = components.findByName('java')
132         if (component instanceof AdhocComponentWithVariants) {
133             return (AdhocComponentWithVariants) component
134         }
135         null
136     }
137 
138     static boolean isGradle6Compatible() {
139         GradleVersion.current() >= GradleVersion.version('6.0')
140     }
141 
142     static void registerJarVariant(String feature, String classifier, TaskProvider<Jar> jarProvider, Project project) {
143         if (isGradle6Compatible()) {
144             Configuration variant = project.configurations.maybeCreate(classifier + 'Elements')
145             variant.visible = false
146             variant.description = feature + ' elements for main.'
147             variant.canBeResolved = false
148             variant.canBeConsumed = true
149             variant.attributes.attribute(Usage.USAGE_ATTRIBUTE, project.objects.named(Usage, Usage.JAVA_RUNTIME))
150             variant.attributes.attribute(Category.CATEGORY_ATTRIBUTE, project.objects.named(Category, Category.DOCUMENTATION))
151             variant.attributes.attribute(Bundling.BUNDLING_ATTRIBUTE, project.objects.named(Bundling, Bundling.EXTERNAL))
152             variant.attributes.attribute(DocsType.DOCS_TYPE_ATTRIBUTE, project.objects.named(DocsType, classifier))
153 
154             variant.outgoing.artifact(new LazyPublishArtifact(jarProvider))
155             AdhocComponentWithVariants component = findJavaComponent(project.components)
156             if (component != null) {
157                 component.addVariantsFromConfiguration(variant, new JavaConfigurationVariantMapping('runtime', true))
158             }
159         }
160     }
