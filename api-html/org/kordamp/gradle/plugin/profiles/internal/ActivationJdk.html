038  *
039  * @author Andres Almiray
040  * @since 0.35.0
041  */
042 @CompileStatic
043 class ActivationJdk implements Activation {
044     final Property<String> version
045 
046     @Inject
047     ActivationJdk(ObjectFactory objects) {
048         version = objects.property(String)
049     }
050 
051     boolean isActive(Project project) {
052         if (!version.present) {
053             throw new IllegalStateException("No value for 'jdk' has been set.")
054         }
055 
056         ArtifactVersion detectedVersion = detectVersion(project)
057         String requiredVersionRange = version.get()
058 
059         if (isNotBlank(requiredVersionRange)) {
060             VersionRange vr = null
061             String msg = 'Detected JDK Version: ' + detectedVersion
062 
063             // short circuit check if the strings are exactly equal
064             if (detectedVersion.toString().equals(requiredVersionRange)) {
065                 project.logger.info(msg + ' is allowed in the range ' + requiredVersionRange + '.')
066             } else {
067                 try {
068                     vr = VersionRange.createFromVersionSpec(requiredVersionRange)
069 
070                     if (containsVersion(vr, detectedVersion)) {
071                         project.logger.info(msg + ' is allowed in the range ' + requiredVersionRange + '.')
072                         return true
073                     } else {
074                         project.logger.info(msg + ' is not in the allowed range ' + vr + '.')
075                         return false
076                     }
077                 } catch (InvalidVersionSpecificationException e) {
078                     project.logger.warn('The requested JDK version '
079                         + requiredVersionRange + ' is invalid.', e)
080                     false
081                 }
082             }
083         }
084 
085         false
086     }
087 
088     private ArtifactVersion detectVersion(Project project) {
089         String javaVersion = System.getProperty('java.version')
090 
091         project.logger.info("Detected Java String: '" + javaVersion + "'")
092         javaVersion = normalizeJDKVersion(javaVersion)
093         project.logger.info("Normalized Java String: '" + javaVersion + "'")
094 
095         ArtifactVersion detectedJdkVersion = new DefaultArtifactVersion(javaVersion)
096 
097         project.logger.info("Parsed Version: Major: " + detectedJdkVersion.majorVersion + " Minor: "
098             + detectedJdkVersion.minorVersion + " Incremental: " + detectedJdkVersion.incrementalVersion
099             + " Build: " + detectedJdkVersion.buildNumber + " Qualifier: " + detectedJdkVersion.qualifier)
100 
101         detectedJdkVersion
102     }
103 
104     static Map<String, String> detectedVersionAsMap() {
105         String javaVersion = System.getProperty('java.version')
106         javaVersion = normalizeJDKVersion(javaVersion)
107         ArtifactVersion detectedJdkVersion = new DefaultArtifactVersion(javaVersion)
108 
109         [
110             version    : javaVersion,
111             major      : String.valueOf(detectedJdkVersion.majorVersion),
112             minor      : String.valueOf(detectedJdkVersion.minorVersion),
113             incremental: String.valueOf(detectedJdkVersion.incrementalVersion),
114             build      : String.valueOf(detectedJdkVersion.buildNumber),
115             qualifier  : String.valueOf(detectedJdkVersion.qualifier)
116         ]
117     }
118 
119     private static String normalizeJDKVersion(String theJdkVersion) {
120         theJdkVersion = theJdkVersion.replaceAll('_|-', '.')
121         String[] tokenArray = split(theJdkVersion, '.')
122         List<String> tokens = Arrays.asList(tokenArray)
123         StringBuffer buffer = new StringBuffer(theJdkVersion.length())
124 
125         Iterator<String> iter = tokens.iterator()
126         for (int i = 0; i < tokens.size() && i < 4; i++) {
127             String section = iter.next()
128             section = section.replaceAll('[^0-9]', '')
129 
130             if (isNotBlank(section)) {
131                 buffer.append(Integer.parseInt(section))
132 
133                 if (i != 2) {
134                     buffer.append('.')
135                 } else {
136                     buffer.append('-')
137                 }
138             }
139         }
140 
141         String version = buffer.toString()
142         version = stripEnd(version, '-')
143         return stripEnd(version, '.')
144     }
145 
146     static boolean containsVersion(VersionRange allowedRange, ArtifactVersion theVersion) {
147         boolean matched = false
148         ArtifactVersion recommendedVersion = allowedRange.getRecommendedVersion()
149         if (recommendedVersion == null) {
150             List<Restriction> restrictions = allowedRange.getRestrictions()
151             for (Restriction restriction : restrictions) {
152                 if (restriction.containsVersion(theVersion)) {
153                     matched = true
154                     break
155                 }
156             }
157         } else {
158             // only singular versions ever have a recommendedVersion
159             @SuppressWarnings('unchecked')
160             int compareTo = recommendedVersion.compareTo(theVersion)
161             matched = (compareTo <= 0)
162         }
163         return matched
164     }
