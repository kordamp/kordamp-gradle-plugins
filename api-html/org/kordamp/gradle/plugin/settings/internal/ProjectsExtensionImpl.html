022 import org.gradle.api.Action
023 import org.gradle.api.Project

027 import org.gradle.api.logging.Logging

029 import org.gradle.api.provider.ListProperty
030 import org.gradle.api.provider.Property

035 /**
036  * @author Andres Almiray
037  * @since 0.35.0
038  */
039 @CompileStatic
040 class ProjectsExtensionImpl implements ProjectsExtension {
041     private static final Logger LOG = Logging.getLogger(Project)
042 
043     final PluginsSpecImpl plugins = new PluginsSpecImpl()
044 
045     final Property<String> layout
046     final Property<Boolean> enforceNamingConvention
047     final ListProperty<String> directories
048     final ListProperty<String> excludes
049     final Property<String> prefix
050     final Property<String> suffix
051     final Property<String> fileNameTransformation
052 
053     final Set<DirectorySpecImpl> directoryConditions = []
054     final Set<PathSpecImpl> pathConditions = []
055 
056     private final Settings settings
057 
058     ProjectsExtensionImpl(Settings settings, ObjectFactory objects) {
059         this.settings = settings
060         layout = objects.property(String).convention('two-level')
061         enforceNamingConvention = objects.property(Boolean).convention(true)
062         directories = objects.listProperty(String).convention([])
063         excludes = objects.listProperty(String).convention([])
064         prefix = objects.property(String)
065         suffix = objects.property(String)
066         fileNameTransformation = objects.property(String)
067 
068         settings.gradle.addBuildListener(new BuildAdapter() {
069             @Override
070             void settingsEvaluated(Settings s) {
071                 if ('two-level' == layout.get().toLowerCase()) {
072                     processTwoLevelLayout()
073                 } else if ('multi-level' == layout.get().toLowerCase()) {
074                     processMultiLevelLayout()
075                 } else if ('standard' == layout.get().toLowerCase()) {
076                     processStandardLayout()
077                 } else if ('explicit' == layout.get().toLowerCase()) {
078                     processExplicitDirectoriesAndPaths()
079                 } else {
080                     LOG.warn "Unknown project layout '${layout.orNull}'. No subprojects will be added."
081                 }
082             }
083 
084             @Override
085             void projectsLoaded(Gradle gradle) {
086                 applyPlugins(gradle.rootProject)
087             }
088         })
089     }
090 
091     @Override
092     void plugins(@DelegatesTo(strategy = Closure.DELEGATE_FIRST, value = PluginsSpec) Closure<Void> action) {
093         ConfigureUtil.configure(action, plugins)
094     }
095 
096     @Override
097     void plugins(Action<? super PluginsSpec> action) {
098         action.execute(plugins)
099     }
100     @Override
101     DirectorySpec includeFromDir(String dir) {
102         DirectorySpecImpl spec = new DirectorySpecImpl(dir)
103         directoryConditions << spec
104         spec
105     }
106 
107     @Override
108     PathSpec includeFromPath(String path) {
109         PathSpecImpl spec = new PathSpecImpl(path)
110         pathConditions << spec
111         spec
112     }
113 
114     private void applyPlugins(Project rootProject) {
115         plugins.apply(rootProject)
116     }
117 
118     private void processExplicitDirectoriesAndPaths() {
119         for (DirectorySpecImpl spec : directoryConditions) {
120             if (!spec.applies()) continue
121             File parentDir = new File(settings.rootDir, spec.dir)
122             if (!parentDir.exists()) {
123                 LOG.info "Skipping ${parentDir} as it does not exist."
124                 continue
125             }
126 
127             excludes.addAll(spec.excludes)
128             doProcessTwoLevelLayout(parentDir)
129         }
130 
131         for (PathSpecImpl spec : pathConditions) {
132             if (!spec.applies()) continue
133             File projectDir = new File(settings.rootDir, spec.path)
134             if (!projectDir.exists()) {
135                 LOG.info "Skipping ${projectDir} as it does not exist."
136                 continue
137             }
138             includeProject(spec.path)
139         }
140     }
141 
142     private void processTwoLevelLayout() {
143         if (directories.get()) {
144             for (String parentDirName : directories.get()) {
145                 File parentDir = new File(settings.rootDir, parentDirName)
146                 if (!parentDir.exists()) {
147                     LOG.info "Skipping ${parentDir} as it does not exist."
148                     continue
149                 }
150 
151                 doProcessTwoLevelLayout(parentDir)
152             }
153         } else {
154             settings.settingsDir.eachDir { File parentDir ->
155                 doProcessTwoLevelLayout(parentDir)
156             }
157         }
158     }
159 
160     private void processMultiLevelLayout() {
161         for (String path : directories.get()) {
162             File projectDir = new File(settings.rootDir, path)
163             if (!projectDir.exists()) {
164                 LOG.info "Skipping ${projectDir} as it does not exist."
165                 continue
166             }
167             includeProject(path)
168         }
169     }
170 
171     private void processStandardLayout() {
172         settings.rootDir.eachDir { File projectDir ->
173             if (excludes.get().contains(projectDir.name)) return
174 
175             File buildFile = resolveBuildFile(projectDir, projectDir.name)
176             doIncludeProject(settings.rootDir, projectDir.name, buildFile)
177         }
178     }
179 
180     private void doProcessTwoLevelLayout(File parentDir) {
181         parentDir.eachDir { File projectDir ->
182             if (excludes.get().contains(projectDir.name)) return
183 
184             File buildFile = resolveBuildFile(projectDir, projectDir.name)
185             doIncludeProject(parentDir, projectDir.name, buildFile)
186         }
187     }
188 
189     private void includeProject(File parentDir, String projectDirName) {
190         File buildFile = resolveBuildFile(parentDir, projectDirName)
191         doIncludeProject(settings.rootDir, parentDir.name, buildFile)
192     }
193 
194     private void includeProject(File projectDir) {
195         includeProject(projectDir.parentFile, projectDir.name)
196     }
197 
198     private void includeProject(String projectPath) {
199         String[] parts = projectPath.split(File.separator)
200         String projectDirName = projectPath
201         String projectName = parts[-1]
202 
203         if (excludes.get().contains(projectName)) return
204 
205         File projectDir = new File(projectDirName)
206 
207         assert projectDir.isDirectory()
208 
209         File buildFile = resolveBuildFile(projectDir, projectName)
210         if (buildFile && buildFile.exists()) {
211             settings.include(projectName)
212             settings.project(':' + projectName).projectDir = projectDir
213             settings.project(':' + projectName).buildFileName = buildFile.name
214             LOG.info("[settings] Including project :${projectName} -> ${buildFile.absolutePath}")
215         }
216     }
217 
218     private File resolveBuildFile(File projectDir, String projectName) {
219         if (enforceNamingConvention.get()) {
220             if ('add'.equalsIgnoreCase(fileNameTransformation.orNull)) {
221                 if (!isBlank(prefix.orNull)) {
222                     projectName = prefix.get() + projectName
223                 }
224                 if (!isBlank(suffix.orNull)) {
225                     projectName += suffix.get()
226                 }
227             } else if ('remove'.equalsIgnoreCase(fileNameTransformation.orNull)) {
228                 if (!isBlank(prefix.get())) {
229                     projectName -= prefix.get()
230                 }
231                 if (!isBlank(suffix.get())) {
232                     projectName -= suffix.get()
233                 }
234             }
235         }
236 
237         for (String fileName : [
238             projectName + '.gradle'.toString(),
239             'build.gradle',
240             projectName + '.gradle.kts'.toString(),
241             'build.gradle.kts']) {
242             File buildFile = new File(projectDir, fileName)
243             if (buildFile.exists()) {
244                 return buildFile
245             }
246 
247         }
248         null
249     }
250 
251     private void doIncludeProject(File parentDir, String projectDirName, File buildFile) {
252         if (buildFile && buildFile.exists()) {
253             File projectDir = new File(parentDir, projectDirName)
254 
255             settings.include(projectDirName)
256             settings.project(':' + projectDirName).projectDir = projectDir
257             settings.project(':' + projectDirName).buildFileName = buildFile.name
258             LOG.info("[settings] Including project :${projectDirName} -> ${buildFile.absolutePath}")
259         }
260     }
261 
262     private boolean isBlank(String str) {
263         if (str == null || str.length() == 0) {
264             return true
265         }
266         for (char c : str.toCharArray()) {
267             if (!Character.isWhitespace(c)) {
268                 return false
269             }
270         }
271 
272         return true
273     }
