041 @CacheableTask
042 @CompileStatic
043 class AggregateSourceStatsReportTask extends DefaultTask {
044     @Optional @Input List<String> formats = []
045 
046     @PathSensitive(PathSensitivity.RELATIVE)
047     @Optional @OutputDirectory File reportDir
048 
049     @Internal
050     File xmlReport
051 
052     @TaskAction
053     @CompileDynamic
054     void computeAggregate() {
055         if (!reportDir) reportDir = project.file("${project.reporting.baseDir.path}/stats")
056         if (!xmlReport) xmlReport = project.file("${reportDir}/${project.name}-all.xml")
057         reportDir.mkdirs()
058 
059         Map<String, Map<String, Integer>> stats = [:]
060 
061         project.subprojects.each { subprj ->
062             subprj.tasks.withType(SourceStatsTask) { SourceStatsTask t ->
063                 if (!t.enabled || !t.xmlReport.exists()) return
064 
065                 def xml = new XmlSlurper().parse(t.xmlReport)
066                 xml.category.each { category ->
067                     String n = category.@name.text() ? category.@name.text() : 'Totals'
068                     Map<String, Integer> map = stats.get(n, [fileCount: 0, locCount: 0])
069                     map.fileCount += category.fileCount.text().toInteger()
070                     map.locCount += category.loc.text().toInteger()
071                 }
072             }
073         }
074 
075         Map totals = stats.remove('Totals')
076         stats = stats.sort { it.key }
077 
078         if (stats) {
079             int max = 0
080             stats.keySet().each { max = Math.max(max, it.size()) }
081             max = Math.max(max, 22)
082 
083             output(stats, max, totals.fileCount.toString(), totals.locCount.toString(), new PrintWriter(System.out))
084             xmlOutput(stats, totals.fileCount.toString(), totals.locCount.toString())
085             if (HTML in formats) htmlOutput(stats, totals.fileCount.toString(), totals.locCount.toString())
086             if (TXT in formats) output(stats, max, totals.fileCount.toString(), totals.locCount.toString(), new PrintWriter(getOutputFile(TXT)))
087         }
088     }
089 
090     private void output(Map<String, Map<String, Object>> stats, int max, String totalFiles, String totalLOC, Writer out) {
091         int padFiles = Math.max(totalFiles.toString().length(), 7)
092         int padLocs = Math.max(totalLOC.toString().length(), 7)
093 
094         out.println '    +-' + ('-' * max) + '-+---------+---------+'
095         out.println '    | ' + 'Name'.padRight(max, ' ') + ' | ' +
096             'Files'.padLeft(padFiles, ' ') + ' | ' +
097             'LOC'.padLeft(padLocs, ' ') + ' |'
098         out.println '    +-' + ('-' * max) + '-+---------+---------+'
099 
100         stats.each { category, info ->
101             out.println '    | ' +
102                 category.padRight(max, ' ') + ' | ' +
103                 info.fileCount.toString().padLeft(padFiles, ' ') + ' | ' +
104                 info.locCount.toString().padLeft(padLocs, ' ') + ' |'
105         }
106 
107         out.println '    +-' + ('-' * max) + '-+---------+---------+'
108         out.println '    | ' + 'Totals'.padRight(max, ' ') + ' | ' + totalFiles.padLeft(padFiles, ' ') + ' | ' + totalLOC.padLeft(padLocs, ' ') + ' |'
109         out.println '    +-' + ('-' * max) + '-+---------+---------+\n'
110 
111         out.flush()
112     }

114     @CompileDynamic
115     private void xmlOutput(Map<String, Map<String, Object>> stats, String totalFiles, String totalLOC) {
116         new MarkupBuilder(new FileWriter(getOutputFile(XML))).stats {
117             stats.each { c, info ->
118                 category(name: c) {
119                     fileCount(info.fileCount.toString())
120                     loc(info.locCount.toString())
121                 }
122             }
123             category {
124                 name('Total')
125                 fileCount(totalFiles)
126                 loc(totalLOC)
127             }
128         }
129     }
130 
131     @CompileDynamic
132     private void htmlOutput(Map<String, Map<String, Object>> stats, String totalFiles, String totalLOC) {
133         int i = 0
134         new MarkupBuilder(new FileWriter(getOutputFile(HTML))).html {
135             table(border: 1) {
136                 tr {
137                     th('Name')
138                     th(align: 'right', 'Files')
139                     th(align: 'right', 'LOC')
140                 }
141                 stats.each { type, info ->
142                     tr(style: (i++) % 2 ? 'background-color:lightblue' : 'background-color:FFF') {
143                         td(type)
144                         td(align: 'right', info.fileCount.toString())
145                         td(align: 'right', info.locCount.toString())
146                     }
147                 }
148                 tr(style: 'background-color:lightgreen') {
149                     b {
150                         td('Total')
151                         td(align: 'right', totalFiles)
152                         td(align: 'right', totalLOC)
153                     }
154                 }
155             }
156         }
157     }
158 
159     private getOutputFile(String suffix) {
160         reportDir.mkdirs()
161         new File(reportDir, project.name + '-all.' + suffix)
162     }
