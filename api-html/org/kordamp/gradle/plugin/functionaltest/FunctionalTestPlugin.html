018 package org.kordamp.gradle.plugin.functionaltest
019 
020 import groovy.transform.CompileStatic
021 import org.gradle.api.Action
022 import org.gradle.api.Project
023 import org.gradle.api.tasks.SourceSet
024 import org.gradle.api.tasks.SourceSetContainer
025 import org.gradle.api.tasks.testing.TestReport
026 import org.kordamp.gradle.plugin.AbstractKordampPlugin
027 import org.kordamp.gradle.plugin.base.BasePlugin
028 import org.kordamp.gradle.plugin.test.tasks.FunctionalTest
029 
030 import static org.kordamp.gradle.PluginUtils.resolveSourceSets
031 import static org.kordamp.gradle.PluginUtils.supportsApiConfiguration
032 
033 /**
034  * @author Andres Almiray
035  * @since 0.2.0
036  */
037 class FunctionalTestPlugin extends AbstractKordampPlugin {
038     Project project
039 
040     @CompileStatic
041     void apply(Project project) {
042         this.project = project
043 
044         if (hasBeenVisited(project)) {
045             return
046         }
047         setVisited(project, true)
048 
049         BasePlugin.applyIfMissing(project)
050 
051         project.pluginManager.withPlugin('java-base') {
052             createSourceSetsIfNeeded(project, 'java')
053             createConfigurationsIfNeeded(project)
054             createTasksIfNeeded(project)
055         }
056 
057         project.pluginManager.withPlugin('groovy-base') {
058             createSourceSetsIfNeeded(project, 'groovy')
059             createConfigurationsIfNeeded(project)
060             createTasksIfNeeded(project)
061         }
062 
063         project.pluginManager.withPlugin('org.jetbrains.kotlin.jvm') {
064             createSourceSetsIfNeeded(project, 'kotlin')
065             createConfigurationsIfNeeded(project)
066             createTasksIfNeeded(project)
067         }
068 
069         project.pluginManager.withPlugin('scala-base') {
070             createSourceSetsIfNeeded(project, 'scala')
071             createConfigurationsIfNeeded(project)
072             createTasksIfNeeded(project)
073         }
074 
075         project.afterEvaluate {
076             adjustSourceSets(project)
077             adjustConfigurations(project)
078             adjustTaskDependencies(project)
079         }
080     }
081 
082     private void adjustSourceSets(Project project) {
083         SourceSet sourceSet = resolveSourceSets(project).functionalTest
084         sourceSet.compileClasspath += resolveSourceSets(project).main.output
085         sourceSet.compileClasspath += project.configurations.compileClasspath
086         sourceSet.runtimeClasspath += sourceSet.compileClasspath
087 
088         project.configurations.findByName('functionalTestCompileClasspath')
089                 .extendsFrom(project.configurations.compileClasspath)
090         project.configurations.findByName('functionalTestRuntimeClasspath')
091                 .extendsFrom(project.configurations.compileClasspath)
092     }
093 
094     @CompileStatic
095     private void adjustConfigurations(Project project) {
096         String compileSuffix = 'Compile'
097         String runtimeSuffix = 'Runtime'
098         if (supportsApiConfiguration(project)) {
099             compileSuffix = 'Implementation'
100             runtimeSuffix = 'RuntimeOnly'
101         }
102 
103         project.configurations.findByName('functionalTest' + compileSuffix)
104                 .extendsFrom project.configurations.findByName(compileSuffix.uncapitalize())
105         project.configurations.findByName('functionalTest' + runtimeSuffix)
106                 .extendsFrom project.configurations.findByName(runtimeSuffix.uncapitalize())
107     }
108 
109     @CompileStatic
110     private void adjustTaskDependencies(Project project) {
111         SourceSet sourceSet = ((SourceSetContainer) resolveSourceSets(project)).findByName('functionalTest')
112         FunctionalTest functionalTest = (FunctionalTest) project.tasks.findByName('functionalTest')
113         functionalTest.classpath = sourceSet.runtimeClasspath
114         TestReport functionalTestReport = (TestReport) project.tasks.findByName('functionalTestReport')
115         functionalTest.dependsOn project.tasks.findByName('jar')
116         functionalTest.mustRunAfter project.tasks.findByName('test')
117         functionalTest.finalizedBy functionalTestReport
118         functionalTestReport.reportOn functionalTest.binResultsDir
119         functionalTestReport.dependsOn functionalTest
120         project.tasks.findByName('check').dependsOn functionalTestReport
121     }
122 
123     private void createConfigurationsIfNeeded(Project project) {
124         String compileSuffix = 'Compile'
125         String runtimeSuffix = 'Runtime'
126         if (supportsApiConfiguration(project)) {
127             compileSuffix = 'Implementation'
128             runtimeSuffix = 'RuntimeOnly'
129         }
130 
131         project.configurations.maybeCreate('functionalTest' + compileSuffix)
132         project.configurations.maybeCreate('functionalTest' + runtimeSuffix)
133     }
134 
135     private void createSourceSetsIfNeeded(Project project, String sourceSetName) {
136         SourceSet sourceSet = project.sourceSets.maybeCreate('functionalTest')
137         if (project.file('src/functional-test/' + sourceSetName).exists()) {
138             sourceSet[sourceSetName].srcDirs project.file('src/functional-test/' + sourceSetName)
139         }
140         if (project.file('src/functional-test/resources').exists()) {
141             sourceSet.resources.srcDir project.file('src/functional-test/resources')
142         }
143     }
144 
145     private void createTasksIfNeeded(Project project) {
146         if (!project.tasks.findByName('functionalTest')) {
147             project.tasks.register('functionalTest', FunctionalTest,
148                     new Action<FunctionalTest>() {
149                         @Override
150                         void execute(FunctionalTest t) {
151                             t.group = 'Verification'
152                             t.description = 'Runs the functional tests.'
153                             t.testClassesDirs = resolveSourceSets(project).functionalTest.output.classesDirs
154                             t.classpath = resolveSourceSets(project).functionalTest.runtimeClasspath
155                             t.reports.html.enabled = false
156                             t.forkEvery = Runtime.runtime.availableProcessors()
157 
158                             t.testLogging {
159                                 events 'passed', 'skipped', 'failed'
160                             }
161                         }
162                     })
163         }
164 
165         if (!project.tasks.findByName('functionalTestReport')) {
166             project.tasks.register('functionalTestReport', TestReport,
167                     new Action<TestReport>() {
168                         @Override
169                         void execute(TestReport t) {
170                             t.group = 'Reporting'
171                             t.description = 'Generates a report on functional tests.'
172                             t.destinationDir = project.file("${project.reporting.baseDir.path}/functional-tests")
173                         }
174                     })
175         }
176     }
177 
178     @CompileStatic
179     static void applyIfMissing(Project project) {
180         if (!project.plugins.findPlugin(FunctionalTestPlugin)) {
181             project.pluginManager.apply(FunctionalTestPlugin)
182         }
183     }
184 }
