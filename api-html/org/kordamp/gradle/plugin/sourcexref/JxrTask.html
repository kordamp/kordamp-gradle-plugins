042 /**
043  * @author Andres Almiray
044  * @since 0.7.0
045  */
046 @CacheableTask
047 @CompileStatic
048 class JxrTask extends DefaultTask {
049     @OutputDirectory File outputDirectory
050 
051     @PathSensitive(PathSensitivity.RELATIVE)
052     @InputFiles FileCollection sourceDirs
053 
054     @Input @Optional String templateDir
055     @Input @Optional String inputEncoding
056     @Input @Optional String outputEncoding
057     @Input @Optional String windowTitle
058     @Input @Optional String docTitle
059     @Input @Optional String bottom = ''
060     @Input @Optional String stylesheet
061     @Input @Optional List<String> includes = []
062     @Input @Optional List<String> excludes = []
063     @Input @Optional JavaVersion javaVersion
064 
065     JxrTask() {
066         windowTitle = "${project.name} ${project.version} Reference"
067         docTitle = "${project.name} ${project.version} Reference"
068     }
069 
070     String getOutputEncoding() {
071         outputEncoding ?: 'UTF-8'
072     }
073 
074     @TaskAction
075     void xref() {
076         JXR jxr = new JXR()
077         jxr.setDest(Paths.get(outputDirectory.absolutePath))
078 
079         if (isBlank(inputEncoding)) {
080             String platformEncoding = System.getProperty('file.encoding')
081             logger.warn("File encoding has not been set, using platform encoding " + platformEncoding
082                 + ", i.e. build is platform dependent!")
083             jxr.setInputEncoding(platformEncoding)
084         } else {
085             jxr.setInputEncoding(inputEncoding)
086         }
087 
088         jxr.outputEncoding = getOutputEncoding()
089         jxr.locale = Locale.default
090         jxr.log = new LogWrapper()
091         jxr.revision = 'HEAD' // TODO
092 
093         if (excludes) {
094             jxr.excludes = excludes.toArray(new String[excludes.size()])
095         }
096         if (includes) {
097             jxr.includes = includes.toArray(new String[includes.size()])
098         }
099 
100         ClassLoader contextClassLoader = Thread.currentThread().contextClassLoader
101         try {
102             Thread.currentThread().contextClassLoader = JxrTask.class.classLoader
103             jxr.xref(resolveSourceDirs(), resolveTemplatesDir(), windowTitle, docTitle, bottom)
104         }
105         finally {
106             Thread.currentThread().contextClassLoader = contextClassLoader
107         }
108 
109         copyResources(outputDirectory.absolutePath)
110     }
111 
112     @CompileDynamic
113     List<String> resolveSourceDirs() {
114         sourceDirs.files.flatten().findAll { it.exists() }*.absolutePath
115     }
116 
117     String resolveTemplatesDir() {
118         JavaVersion javadocTemplatesVersion = resolveJavadocTemplatesVersion()
119         // Check if overridden
120         if (isBlank(templateDir)) {
121             if (javadocTemplatesVersion.java8Compatible) {
122                 return 'templates/jdk8'
123             } else if (javadocTemplatesVersion.java7Compatible) {
124                 return 'templates/jdk7'
125             } else if (javadocTemplatesVersion.java5Compatible) {
126                 return 'templates/jdk4'
127             } else {
128                 logger.warn('Unsupported javadocVersion: ' + javadocTemplatesVersion + '. Fallback to original')
129                 return 'templates'
130             }
131         }
132         // use value specified by user
133         return templateDir
134     }
135 
136     JavaVersion resolveJavadocTemplatesVersion() {
137         javaVersion ?: JavaVersion.current()
138     }
139 
140     private void copyResources(String dir) {
141         if (isNotBlank(stylesheet)) {
142             File stylesheetFile = new File(stylesheet)
143             File destStylesheetFile = new File(dir, 'stylesheet.css')
144 
145             try {
146                 if (stylesheetFile.isAbsolute()) {
147                     Files.copy(Paths.get(stylesheetFile.toURI()), Paths.get(destStylesheetFile.toURI()))
148                 } else {
149                     URL stylesheetUrl = JxrTask.class.classLoader.getResource(stylesheet)
150                     destStylesheetFile.parentFile.mkdirs()
151                     destStylesheetFile.text = stylesheetUrl.text
152                 }
153             }
154             catch (IOException e) {
155                 logger.warn('An error occured while copying the stylesheet to the target directory', e)
156             }
157         } else {
158             JavaVersion javadocTemplatesVersion = resolveJavadocTemplatesVersion()
159 
160             if (javadocTemplatesVersion.java8Compatible) {
161                 copyResources(dir, 'jdk8/', 'stylesheet.css')
162             } else if (javadocTemplatesVersion.java7Compatible) {
163                 String[] jdk7Resources = [
164                     'stylesheet.css',
165                     'resources/background.gif',
166                     'resources/tab.gif',
167                     'resources/titlebar.gif',
168                     'resources/titlebar_end.gif'
169                 ] as String[]
170                 copyResources(dir, 'jdk7/', jdk7Resources)
171             } else if (javadocTemplatesVersion.java6Compatible) {
172                 copyResources(dir, 'jdk6/', 'stylesheet.css')
173             } else if (javadocTemplatesVersion.java5Compatible) {
174                 copyResources(dir, 'jdk4/', 'stylesheet.css')
175             } else {
176                 // Fallback to the original stylesheet
177                 copyResources(dir, '', 'stylesheet.css')
178             }
179         }
180     }
181 
182     private void copyResources(String dir, String sourceFolder, String... files) {
183         try {
184             for (String file : files) {
185                 URL resourceUrl = JxrTask.class.classLoader.getResource(sourceFolder + file)
186                 File destResourceFile = new File(dir, file)
187                 destResourceFile.parentFile.mkdirs()
188                 destResourceFile.text = resourceUrl.text
189             }
190         }
191         catch (IOException e) {
192             logger.warn('An error occured while copying the resource to the target directory', e)
193         }
194     }
