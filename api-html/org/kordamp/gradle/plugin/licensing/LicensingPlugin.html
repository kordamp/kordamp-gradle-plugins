270     @CompileDynamic
271     private void configureLicenseExtension(Project project) {
272         ProjectConfigurationExtension config = resolveConfig(project)
273         setEnabled(config.licensing.enabled)
274 
275         if (!enabled || config.licensing.empty) {
276             project.tasks.withType(LicenseCheck).each { it.enabled = false }
277             project.tasks.withType(LicenseFormat).each { it.enabled = false }
278             return
279         }
280 
281         License lic = config.licensing.allLicenses()[0]
282         if (config.licensing.allLicenses().size() > 1) {
283             lic = config.licensing.allLicenses().find {
284                 it.primary
285             } ?: config.licensing.allLicenses()[0]
286         }
287 
288         LicenseExtension licenseExtension = project.extensions.findByType(LicenseExtension)
289 
290         licenseExtension.header = project.rootProject.file('gradle/LICENSE_HEADER')
291         licenseExtension.strictCheck = true
292         licenseExtension.mapping {
293             java = 'SLASHSTAR_STYLE'
294             groovy = 'SLASHSTAR_STYLE'
295             kt = 'SLASHSTAR_STYLE'
296             scala = 'SLASHSTAR_STYLE'
297             gradle = 'SLASHSTAR_STYLE'
298             kts = 'SLASHSTAR_STYLE'
299             yml = 'SCRIPT_STYLE'
300             toml = 'SCRIPT_STYLE'
301         }
302         licenseExtension.ext.project = project.name
303         licenseExtension.ext {
304             projectName = config.info.name
305             copyrightYear = config.info.copyrightYear
306             author = config.info.getAuthors().join(', ')
307             license = lic.licenseId?.spdx()
308         }
309         licenseExtension.exclude '**/*.png'
310         licenseExtension.exclude '**/*.gif'
311         licenseExtension.exclude '**/*.jpg'
312         licenseExtension.exclude '**/*.jpeg'
313         licenseExtension.exclude '**/*.json'
314         licenseExtension.exclude 'META-INF/services/*'
315 
316         project.tasks.withType(LicenseCheck, new Action<LicenseCheck>() {
317             @Override
318             void execute(LicenseCheck t) {
319                 String sourceSetName = t.name[7..-1].uncapitalize()
320                 t.enabled = !config.licensing.excludedSourceSets.contains(sourceSetName)
321             }
322         })
323 
324         project.tasks.withType(LicenseFormat, new Action<LicenseFormat>() {
325             @Override
326             void execute(LicenseFormat t) {
327                 String sourceSetName = t.name[7..-1].uncapitalize()
328                 t.enabled = !config.licensing.excludedSourceSets.contains(sourceSetName)
329             }
330         })
331     }

333     private void preConfigureDownloadLicensesExtension(Project project) {
334         DownloadLicensesExtension extension = project.extensions.findByType(DownloadLicensesExtension)
335         extension.aliases = new LinkedHashMap<>(DEFAULT_ALIASES)
336     }

338     @CompileDynamic
339     private void postConfigureDownloadLicensesExtension(Project project) {
340         ProjectConfigurationExtension config = resolveConfig(project)
341 
342         Map<Object, List<Object>> defaultAliases = new LinkedHashMap<>(DEFAULT_ALIASES)
343         config.licensing.licenses.licenses.each { license ->
344             if (license.licenseId && license.aliases) {
345                 LicenseMetadata licenseMetadata = LICENSES_MAP.get(license.licenseId)
346                 if (!licenseMetadata) {
347                     licenseMetadata = new LicenseMetadata(license.name, license.url)
348                     LICENSES_MAP.put(license.licenseId, licenseMetadata)
349                 }
350 
351                 List<Object> aliases = defaultAliases.get(licenseMetadata)
352                 if (!aliases) {
353                     aliases = []
354                     defaultAliases.put(licenseMetadata, aliases)
355                 }
356                 List<String> combined = aliases + license.aliases
357                 combined = combined.unique() - aliases
358                 aliases.addAll(combined)
359             }
360         }
361 
362         DownloadLicensesExtension extension = project.extensions.findByType(DownloadLicensesExtension)
363         extension.aliases = new LinkedHashMap<>(defaultAliases)
364 
365         project.tasks.withType(DownloadLicenses) { DownloadLicenses task ->
366             task.aliases = defaultAliases
367         }
368     }

370     private void configureAggregateLicenseReportTask(Project project) {
371         ProjectConfigurationExtension config = resolveConfig(project)
372 
373         if (!config.licensing.enabled) {
374             return
375         }
376 
377         Set<DownloadLicenses> tasks = new LinkedHashSet<>()
378         project.subprojects.each { prj ->
379             tasks.addAll(prj.tasks.withType(DownloadLicenses))
380         }
381 
382         project.tasks.named('aggregateLicenseReport', AggregateLicenseReportTask,
383             new Action<AggregateLicenseReportTask>() {
384                 @Override
385                 void execute(AggregateLicenseReportTask t) {
386                     t.dependsOn tasks
387                     t.enabled = true
388                 }
389             })
390     }
