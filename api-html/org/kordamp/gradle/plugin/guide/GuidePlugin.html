052     static final String ZIP_GUIDE_TASK_NAME = 'zipGuide'
053     static final String ASCIIDOCTOR = 'asciidoctor'
054     static final String ASCIIDOCTOR_SRC_DIR = 'src/docs/asciidoc'

060     GuidePlugin() {
061         super(Guide.PLUGIN_ID)

064     void apply(Project project) {
065         this.project = project
066 
067         if (hasBeenVisited(project)) {
068             return
069         }
070         setVisited(project, true)
071 
072         BasePlugin.applyIfMissing(project)
073         SourceHtmlPlugin.applyIfMissing(project.rootProject)
074         project.pluginManager.apply(AsciidoctorJPlugin)
075 
076         project.extensions.create('guide', GuideExtension, project)
077 
078         addProjectEvaluatedListener(project, new GuideProjectEvaluatedListener())
079     }
080 
081     static void applyIfMissing(Project project) {
082         if (!project.plugins.findPlugin(GuidePlugin)) {
083             project.pluginManager.apply(GuidePlugin)
084         }
085     }

087     @Named('guide')
088     @DependsOn(['base'])
089     private class GuideProjectEvaluatedListener implements ProjectEvaluatedListener {
090         @Override
091         void projectEvaluated(Project project) {

099     @Named('guide')
100     @DependsOn(['base', 'javadoc', 'groovydoc', 'kotlindoc', 'sourceHtml', 'sourceXref'])
101     private class GuideAllProjectsEvaluatedListener implements AllProjectsEvaluatedListener {
102         @Override
103         void allProjectsEvaluated(Project rootProject) {
104             GuideExtension extension = project.extensions.findByType(GuideExtension)
105 
106             project.tasks.named(CREATE_GUIDE_TASK_NAME, Copy, new Action<Copy>() {
107                 @Override
108                 void execute(Copy t) {
109                     Task task = rootProject.tasks.findByName('aggregateJavadoc')
110                     if (task?.enabled) {
111                         t.dependsOn task
112                         t.from(task.destinationDir) { into extension.javadocApiDir }
113                     }
114 
115                     task = rootProject.tasks.findByName('aggregateGroovydoc')
116                     if (task?.enabled) {
117                         t.dependsOn task
118                         t.from(task.destinationDir) { into extension.groovydocApiDir }
119                     }
120 
121                     task = rootProject.tasks.findByName('aggregateKotlindocHtml')
122                     if (task?.enabled) {
123                         t.dependsOn task
124                         t.from(task.outputDirectory) { into extension.kotlindocApiDir }
125                     }
126 
127                     task = rootProject.tasks.findByName(SourceHtmlPlugin.AGGREGATE_SOURCE_HTML_TASK_NAME)
128                     if (task?.enabled) {
129                         t.dependsOn task
130                         t.from(task.destinationDir) { into extension.sourceHtmlDir }
131                     }
132 
133                     task = rootProject.tasks.findByName('aggregateSourceXref')
134                     if (task?.enabled) {
135                         t.dependsOn task
136                         t.from(task.outputDirectory) { into extension.sourceXrefDir }
137                     }
138                 }
139             })
140         }

143     private void configureAsciidoctorTask(Project project) {
144         project.tasks.named(ASCIIDOCTOR).configure(new Action<AsciidoctorTask>() {
145             @Override
146             void execute(AsciidoctorTask t) {
147                 ProjectConfigurationExtension config = resolveConfig(project.rootProject) ?: resolveConfig(project)
148 
149                 Map attrs = [:]
150                 attrs.putAll(t.attributes)
151                 checkAttribute(attrs, t.attributes, 'toc', 'left')
152                 checkAttribute(attrs, t.attributes, 'doctype', 'book')
153                 checkAttribute(attrs, t.attributes, 'icons', 'font')
154                 checkAttribute(attrs, t.attributes, 'encoding', 'utf-8')
155                 checkAttribute(attrs, t.attributes, 'sectlink', true)
156                 checkAttribute(attrs, t.attributes, 'sectanchors', true)
157                 checkAttribute(attrs, t.attributes, 'numbered', true)
158                 checkAttribute(attrs, t.attributes, 'linkattrs', true)
159                 checkAttribute(attrs, t.attributes, 'imagesdir', IMAGES_DIR)
160                 checkAttribute(attrs, t.attributes, 'linkcss', true)
161                 checkAttribute(attrs, t.attributes, 'source-highlighter', 'coderay')
162                 checkAttribute(attrs, t.attributes, 'coderay-linenums-mode', 'table')
163                 checkAttribute(attrs, t.attributes, 'rootdir', project.rootProject.projectDir.absolutePath)
164                 checkAttribute(attrs, t.attributes, 'project-title', config.info.description)
165                 checkAttribute(attrs, t.attributes, 'project-inception-year', config.info.inceptionYear)
166                 checkAttribute(attrs, t.attributes, 'project-copyright-year', config.info.copyrightYear)
167                 checkAttribute(attrs, t.attributes, 'project-author', config.info.getAuthors().join(', '))
168                 checkAttribute(attrs, t.attributes, 'project-url', config.info.url)
169                 checkAttribute(attrs, t.attributes, 'project-scm', config.info.links.scm)
170                 checkAttribute(attrs, t.attributes, 'project-issue-tracker', config.info.links.issueTracker)
171                 checkAttribute(attrs, t.attributes, 'project-group', project.group)
172                 checkAttribute(attrs, t.attributes, 'project-version', project.version)
173                 checkAttribute(attrs, t.attributes, 'project-name', project.rootProject.name)
174 
175                 checkAttribute(attrs, t.attributes, 'build-by', config.buildInfo.buildBy)
176                 checkAttribute(attrs, t.attributes, 'build-date', config.buildInfo.buildDate)
177                 checkAttribute(attrs, t.attributes, 'build-time', config.buildInfo.buildTime)
178                 checkAttribute(attrs, t.attributes, 'build-revision', config.buildInfo.buildRevision)
179                 checkAttribute(attrs, t.attributes, 'build-jdk', config.buildInfo.buildJdk)
180                 checkAttribute(attrs, t.attributes, 'build-created-by', config.buildInfo.buildCreatedBy)
181 
182                 t.attributes(attrs)
183 
184                 t.sourceDir = project.file(ASCIIDOCTOR_SRC_DIR)
185 
186                 t.sources {
187                     include 'index.adoc'
188                 }
189 
190                 t.resources {
191                     from project.file(ASCIIDOCTOR_RESOURCE_DIR)
192                 }
193             }
194         })
195     }

197     private static void checkAttribute(Map dest, Map src, String key, value) {
198         if (!src.containsKey(key)) dest[key + '@'] = value
199     }

201     private void createGuideTask(Project project) {
202         TaskProvider<Copy> guideTask = project.tasks.register(CREATE_GUIDE_TASK_NAME, Copy,
203             new Action<Copy>() {
204                 @Override
205                 void execute(Copy t) {
206                     t.group = 'Documentation'
207                     t.description = 'Creates an Asciidoctor based guide.'
208                     t.dependsOn project.tasks.named(ASCIIDOCTOR)
209                     t.setDestinationDir(project.file("${project.buildDir}/guide"))
210                     t.from(project.tasks.asciidoctor.outputDir)
211                 }
212             })
213 
214         project.tasks.register(ZIP_GUIDE_TASK_NAME, Zip,
215             new Action<Zip>() {
216                 @Override
217                 void execute(Zip t) {
218                     t.dependsOn guideTask
219                     t.group = 'Documentation'
220                     t.description = 'An archive of the generated guide.'
221                     t.setArchiveBaseName(project.rootProject.name + '-guide')
222                     t.from guideTask.get().destinationDir
223                 }
224             })
225 
226         addAllProjectsEvaluatedListener(project.rootProject,new GuideAllProjectsEvaluatedListener())
227     }

229     private void createInitGuideTask(Project project) {
230         project.tasks.register(INIT_GUIDE_TASK_NAME, DefaultTask, new Action<DefaultTask>() {
231             @Override
232             void execute(DefaultTask t) {
233                 t.group = 'Project Setup'
234                 t.description = 'Initializes directories and files required by the guide.'
235                 t.outputs.dir(ASCIIDOCTOR_SRC_DIR)
236                 t.outputs.dir(ASCIIDOCTOR_RESOURCE_DIR)
237                 t.doFirst {
238                     GuidePlugin.initGuide(project)
239                 }
240             }
241         })
242     }

244     @CompileDynamic
245     private void configurePublishing(Project project) {
246         ProjectConfigurationExtension config = resolveConfig(project)
247         if (!config.docs.guide.publish.enabled) {
248             return
249         }
250 
251         Task createGuideTask = project.tasks.findByName(CREATE_GUIDE_TASK_NAME)
252 
253         project.gitPublish {
254             repoUri = config.info.resolveScmLink()
255             branch = config.docs.guide.publish.branch
256             contents {
257                 from createGuideTask.outputs.files
258             }
259             commitMessage = config.docs.guide.publish.message
260         }
261 
262         project.gitPublishCommit.dependsOn(createGuideTask)
263     }

265     static void initGuide(Project project) {
266         GuideExtension extension = project.extensions.findByType(GuideExtension)
267 
268         project.file(ASCIIDOCTOR_RESOURCE_DIR).mkdirs()
269         project.file("$ASCIIDOCTOR_RESOURCE_DIR/$IMAGES_DIR").mkdirs()
270         File asciidocDir = project.file(ASCIIDOCTOR_SRC_DIR)
271         asciidocDir.mkdirs()
272 
273         touchFile(project.file("${asciidocDir}/_links.adoc"))
274 
275         File index = touchFile(project.file("${asciidocDir}/index.adoc"))
276         if (!index.text) {
277             index.text = """|ifndef::imagesdir[]
278                             |:imagesdir: ../resources/$IMAGES_DIR
279                             |endif::[]
280                             |ifndef::includedir[]
281                             |:includedir: .
282                             |endif::[]
283                             |= {project-title}
284                             |:author: {project-author}
285                             |:revnumber: {project-version}
286                             |:toclevels: 4
287                             |:docinfo1:
288                             |
289                             |include::{includedir}/_links.adoc[]
290                             |
291                             |:leveloffset: 1
292                             |include::{includedir}/introduction.adoc[]
293                             |include::{includedir}/usage.adoc[]
294                             |
295                             |= Links
296                             |
297                             |link:${extension.javadocApiDir}/index.html[Javadoc, window="_blank"]
298                             |
299                             |link:${extension.sourceHtmlDir}/index.html[Source, window="_blank"]""".stripMargin('|')
300         }
301 
302         File introduction = touchFile(project.file("${asciidocDir}/introduction.adoc"))
303         if (!introduction.text) {
304             introduction.text = """|
305                                    |[[_introduction]]
306                                    |= Introduction
307                                    |
308                                    |Lorem ipsum dolor sit amet
309                                    |""".stripMargin('|')
310         }
311 
312         File usage = touchFile(project.file("${asciidocDir}/usage.adoc"))
313         if (!usage.text) {
314             usage.text = """|
315                                    |[[_usage]]
316                                    |= Usage
317                                    |
318                                    |Lorem ipsum dolor sit amet
319                                    |""".stripMargin('|')
320         }
321     }

323     static File touchFile(File file) {
324         file.parentFile.mkdirs()
325         if (!file.exists()) {
326             file.text = ''
327         }
328         file
329     }
