018 package org.kordamp.gradle.plugin.publishing
019 
020 import groovy.transform.CompileStatic
021 import org.gradle.api.publish.Publication
022 import org.gradle.api.publish.PublishingExtension
023 import org.gradle.api.publish.maven.MavenPublication
024 import org.gradle.api.tasks.TaskAction
025 import org.gradle.api.tasks.options.Option
026 import org.kordamp.gradle.plugin.base.tasks.AbstractReportingTask
027 
028 import static org.kordamp.gradle.StringUtils.isNotBlank
029 
030 /**
031  * @author Andres Almiray
032  * @since 0.20.0
033  */
034 @CompileStatic
035 class PublicationSettingsTask extends AbstractReportingTask {
036     private boolean absolute
037     private String publication
038     private Set<String> publications
039 
040     @Option(option = 'absolute', description = "Should paths be printed as absolutes or not. Defaults to 'false'")
041     void setAbsolute(boolean absolute) {
042         this.absolute = absolute
043     }
044 
045     @Option(option = 'publication', description = 'The publication to generate the report for.')
046     void setPublication(String publication) {
047         this.publication = publication
048     }
049 
050     @Option(option = 'publications', description = 'The publications to generate the report for.')
051     void setPublications(String publications) {
052         if (isNotBlank(publications)) {
053             this.publications = (publications.split(',').collect { it.trim() }) as Set
054         }
055     }
056 
057     @TaskAction
058     void report() {
059         PublishingExtension publishing = project.extensions.findByType(PublishingExtension)
060         if (publications) {
061             publications.each { p ->
062                 printPublication(publishing.publications.findByName(p))
063             }
064         } else if (publications) {
065             printPublication(publishing.publications.findByName(publication))
066         } else {
067             publishing.publications.each { Publication p ->
068                 printPublication(p)
069             }
070         }
071     }
072 
073     private void printPublication(Publication publication) {
074         print(publication.name + ':', 0)
075         if (publication instanceof MavenPublication) {
076             doPrintMapEntry('type', 'Maven', 1)
077             MavenPublication p = (MavenPublication) publication
078             doPrintMapEntry('groupId', p.groupId, 1)
079             doPrintMapEntry('artifactId', p.artifactId, 1)
080             doPrintMapEntry('version', p.version, 1)
081             if (p.artifacts.size()) {
082                 print('artifacts:', 1)
083                 int index = 0
084                 p.artifacts.each { artifact ->
085                     print('artifact ' + (index++) + ':', 2)
086                     doPrintMapEntry('file', adjustPath(project.rootProject.projectDir.toString(), artifact.file?.absolutePath), 3)
087                     doPrintMapEntry('classifier', artifact.classifier, 3)
088                     doPrintMapEntry('extension', artifact.extension, 3)
089                 }
090             }
091         }
092         println ' '
093     }
094 
095     private String adjustPath(String rootDir, String path) {
096         if (isNotBlank(path) && !absolute) {
097             return (path - rootDir)[1..-1]
098         }
099         return path
100     }
101 }
