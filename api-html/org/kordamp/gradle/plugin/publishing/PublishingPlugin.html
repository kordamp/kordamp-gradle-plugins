042 /**
043  * Configures artifact publication.
044  *
045  * @author Andres Almiray
046  * @since 0.2.0
047  */
048 class PublishingPlugin implements Plugin<Project> {
049     private static final String VISITED = PublishingPlugin.class.name.replace('.', '_') + '_VISITED'
050 
051     Project project
052 
053     void apply(Project project) {
054         this.project = project
055 
056         if (isRootProject(project)) {
057             if (project.childProjects.size()) {
058                 project.childProjects.values().each {
059                     configureProject(it)
060                 }
061             } else {
062                 configureProject(project)
063             }
064         } else {
065             configureProject(project)
066         }
067     }
068 
069     static void applyIfMissing(Project project) {
070         if (!project.plugins.findPlugin(PublishingPlugin)) {
071             project.plugins.apply(PublishingPlugin)
072         }
073         if (!project.plugins.findPlugin(SigningPlugin)) {
074             project.plugins.apply(SigningPlugin)
075         }
076     }
077 
078     private void configureProject(Project project) {
079         String visitedPropertyName = VISITED + '_' + project.name
080         if (project.findProperty(visitedPropertyName)) {
081             return
082         }
083         project.ext[visitedPropertyName] = true
084 
085         BasePlugin.applyIfMissing(project)
086         BuildInfoPlugin.applyIfMissing(project)
087         SourceJarPlugin.applyIfMissing(project)
088         ApidocPlugin.applyIfMissing(project)
089         JarPlugin.applyIfMissing(project)
090 
091         project.plugins.withType(JavaBasePlugin) {
092             if (!project.plugins.findPlugin(MavenPublishPlugin)) {
093                 project.plugins.apply(MavenPublishPlugin)
094             }
095         }
096 
097         project.afterEvaluate {
098             project.plugins.withType(JavaBasePlugin) {
099                 updatePublications(project)
100             }
101         }
102     }
103 
104     private void updatePublications(Project project) {
105         ProjectConfigurationExtension mergedConfiguration = project.ext.mergedConfiguration
106 
107         if (!mergedConfiguration.publishing.enabled || !project.sourceSets.findByName('main')) {
108             return
109         }
110 
111         Task javadocJar = project.tasks.findByName(JavadocPlugin.JAVADOC_JAR_TASK_NAME)
112         Task groovydocJar = project.tasks.findByName(GroovydocPlugin.GROOVYDOC_JAR_TASK_NAME)
113         Task sourceJar = project.tasks.findByName(SourceJarPlugin.SOURCE_JAR_TASK_NAME)
114 
115         project.publishing {
116             publications {
117                 mainPublication(MavenPublication) {
118                     from project.components.java
119 
120                     if (javadocJar?.enabled) artifact javadocJar
121                     if (groovydocJar?.enabled) artifact groovydocJar
122                     if (sourceJar?.enabled) artifact sourceJar
123 
124                     pom {
125                         name = mergedConfiguration.info.name
126                         description = mergedConfiguration.info.description
127                         url = mergedConfiguration.info.url
128                         inceptionYear = mergedConfiguration.info.inceptionYear
129                         licenses {
130                             mergedConfiguration.license.licenses.forEach { lic ->
131                                 license {
132                                     name = lic.name
133                                     url = lic.url
134                                     distribution = lic.distribution
135                                     if (lic.comments) comments = lic.comments
136                                 }
137                             }
138                         }
139                         if (!isBlank(mergedConfiguration.info.scm.url)) {
140                             scm {
141                                 url = mergedConfiguration.info.scm.url
142                                 if (mergedConfiguration.info.scm.connection) {
143                                     connection = mergedConfiguration.info.scm.connection
144                                 }
145                                 if (mergedConfiguration.info.scm.connection) {
146                                     developerConnection = mergedConfiguration.info.scm.developerConnection
147                                 }
148                             }
149                         } else if (mergedConfiguration.info.links.scm) {
150                             scm {
151                                 url = mergedConfiguration.info.links.scm
152                             }
153                         }
154                         if (!mergedConfiguration.info.organization.isEmpty()) {
155                             organization {
156                                 name = mergedConfiguration.info.organization.name
157                                 url = mergedConfiguration.info.organization.url
158                             }
159                         }
160                         developers {
161                             mergedConfiguration.info.people.forEach { Person person ->
162                                 if ('developer' in person.roles*.toLowerCase()) {
163                                     developer {
164                                         if (person.id) id = person.id
165                                         if (person.name) name = person.name
166                                         if (person.url) url = person.url
167                                         if (person.email) email = person.email
168                                         if (person.organization?.name) organizationName = person.organization.name
169                                         if (person.organization?.url) organizationUrl = person.organization.url
170                                         if (person.roles) roles = person.roles as Set
171                                         if (person.properties) properties.set(person.properties)
172                                     }
173                                 }
174                             }
175                         }
176                         contributors {
177                             mergedConfiguration.info.people.forEach { Person person ->
178                                 if ('contributor' in person.roles*.toLowerCase()) {
179                                     contributor {
180                                         if (person.name) name = person.name
181                                         if (person.url) url = person.url
182                                         if (person.email) email = person.email
183                                         if (person.organization?.name) organizationName = person.organization.name
184                                         if (person.organization?.url) organizationUrl = person.organization.url
185                                         if (person.roles) roles = person.roles as Set
186                                         if (person.properties) properties.set(person.properties)
187                                     }
188                                 }
189                             }
190                         }
191                     }
192                 }
193             }
194 
195             String repositoryName = mergedConfiguration.release ? mergedConfiguration.publishing.releasesRepository : mergedConfiguration.publishing.snapshotsRepository
196             if (!isBlank(repositoryName)) {
197                 Repository repo = mergedConfiguration.info.repositories.getRepository(repositoryName)
198                 if (repo == null) {
199                     throw new IllegalStateException("Repository '${repositoryName}' was not found")
200                 }
201 
202                 repositories {
203                     maven {
204                         url = repo.url
205                         Credentials creds = mergedConfiguration.info.credentials.getCredentials(repo.name)
206                         if (repo.credentials) {
207                             credentials {
208                                 username = repo.credentials.username
209                                 password = repo.credentials.password
210                             }
211                         } else if (creds) {
212                             credentials {
213                                 username = creds.username
214                                 password = creds.password
215                             }
216                         }
217                     }
218                 }
219             }
220         }
221 
222         if (mergedConfiguration.publishing.signing) {
223             project.signing {
224                 sign project.publishing.publications.mainPublication
225             }
226         }
227     }
228 }
