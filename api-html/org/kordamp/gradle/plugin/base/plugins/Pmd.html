018 package org.kordamp.gradle.plugin.base.plugins
019 
020 import groovy.transform.Canonical
021 import groovy.transform.CompileDynamic
022 import groovy.transform.CompileStatic
023 import org.gradle.api.Project
024 import org.gradle.api.file.FileCollection
025 import org.gradle.api.plugins.quality.TargetJdk
026 import org.kordamp.gradle.plugin.base.ProjectConfigurationExtension
027 
028 /**
029  * @author Andres Almiray
030  * @since 0.31.0
031  */
032 @CompileStatic
033 @Canonical
034 class Pmd extends AbstractQualityFeature {
035     static final String PLUGIN_ID = 'org.kordamp.gradle.pmd'
036 
037     FileCollection ruleSetFiles
038     boolean incrementalAnalysis
039     int rulePriority = 5
040 
041     private boolean incrementalAnalysisSet
042 
043     Pmd(ProjectConfigurationExtension config, Project project) {
044         super(config, project, PLUGIN_ID, 'pmd')
045         toolVersion = '6.2.0'
046     }
047 
048     @Override
049     protected void populateMapDescription(Map<String, Object> map) {
050         super.populateMapDescription(map)
051         map.ruleSetFiles = this.ruleSetFiles
052         map.incrementalAnalysis = this.incrementalAnalysis
053         map.rulePriority = this.rulePriority
054     }
055 
056     @Override
057     void normalize() {
058         if (null == ruleSetFiles) {
059             FileCollection files = project.rootProject.files("config/pmd/${project.name}.xml")
060             if (!files.empty) {
061                 files = project.rootProject.files('config/pmd/pmd.xml')
062             }
063             ruleSetFiles = files
064         }
065 
066         super.normalize()
067     }
068 
069     void setIncrementalAnalysis(boolean incrementalAnalysis) {
070         this.incrementalAnalysis = incrementalAnalysis
071         this.incrementalAnalysisSet = true
072     }
073 
074     boolean isIncrementalAnalysisSet() {
075         this.incrementalAnalysisSet
076     }
077 
078     void copyInto(Pmd copy) {
079         super.copyInto(copy)
080         copy.ruleSetFiles = ruleSetFiles
081         copy.@incrementalAnalysis = incrementalAnalysis
082         copy.@incrementalAnalysisSet = incrementalAnalysisSet
083         copy.rulePriority = rulePriority
084     }
085 
086     static void merge(Pmd o1, Pmd o2) {
087         AbstractQualityFeature.merge(o1, o2)
088         o1.setIncrementalAnalysis((boolean) (o1.incrementalAnalysisSet ? o1.incrementalAnalysis : o2.incrementalAnalysis))
089         o1.ruleSetFiles = o1.ruleSetFiles ?: o2.ruleSetFiles
090         o1.rulePriority = o1.rulePriority ?: o2.rulePriority
091     }
092 
093     @CompileDynamic
094     void applyTo(org.gradle.api.plugins.quality.Pmd pmdTask) {
095         String sourceSetName = (pmdTask.name - 'pmd').uncapitalize()
096         sourceSetName = sourceSetName == 'allPmd' ? project.name : sourceSetName
097         sourceSetName = sourceSetName == 'aggregatePmd' ? 'aggregate' : sourceSetName
098         pmdTask.enabled = enabled && !ruleSetFiles.empty && ruleSetFiles.files.every { it.exists() }
099         pmdTask.targetJdk = TargetJdk.VERSION_1_7
100         pmdTask.ruleSetFiles = ruleSetFiles
101         pmdTask.ignoreFailures = ignoreFailures
102         pmdTask.getIncrementalAnalysis().set(incrementalAnalysisSet)
103         pmdTask.rulePriority = rulePriority
104         pmdTask.reports.html.enabled = true
105         pmdTask.reports.xml.enabled = true
106         pmdTask.reports.html.destination = project.layout.buildDirectory.file("reports/pmd/${sourceSetName}.html").get().asFile
107         pmdTask.reports.xml.destination = project.layout.buildDirectory.file("reports/pmd/${sourceSetName}.xml").get().asFile
108     }
109 }
