074     private String resolveJavadocLinks(Object jv) {
075         JavaVersion javaVersion = JavaVersion.current()
076 
077         if (jv instanceof JavaVersion) {
078             javaVersion = (JavaVersion) jv
079         } else if (jv != null) {
080             javaVersion = JavaVersion.toVersion(jv)
081         }
082 
083         if (javaVersion.isJava11Compatible()) {
084             return "https://docs.oracle.com/en/java/javase/${javaVersion.majorVersion}/docs/api/".toString()
085         }
086         return "https://docs.oracle.com/javase/${javaVersion.majorVersion}/docs/api/"
087     }

241         static void merge(AutoLinks o1, AutoLinks o2) {
242             o1.setEnabled((boolean) (o1.enabledSet ? o1.enabled : o2.enabled))
243             CollectionUtils.merge(o1.@excludes, o2?.excludes)
244             CollectionUtils.merge(o1.@configurations, o2?.configurations)
245             CollectionUtils.merge(o1.@offlineLinks, o2?.offlineLinks)
246         }

248         Map<String, Object> toMap() {
249             Map<String, Object> map = new LinkedHashMap<String, Object>(enabled: enabled)
250 
251             if (enabled) {
252                 List<String> cs = new ArrayList<>(configurations)
253                 if (!cs) {
254                     cs = DEFAULT_CONFIGURATIONS
255                 }
256                 map.excludes = excludes
257                 map.configurations = cs
258                 map.offlineLinks = offlineLinks
259             }
260 
261             map
262         }

264         List<String> resolveLinks(Project project) {
265             List<String> links = []
266 
267             if (!enabled) return links
268 
269             List<String> cs = new ArrayList<>(configurations)
270             if (!cs) {
271                 cs = DEFAULT_CONFIGURATIONS
272             }
273 
274             for (String cn : cs) {
275                 Configuration c = project.configurations.findByName(cn)
276                 c?.dependencies?.each { Dependency dep ->
277                     if (dep instanceof ProjectDependency) {
278                         ProjectDependency pdep = (ProjectDependency) dep
279                         String packageListLoc = calculateLocalJavadocLink(config.project, pdep.dependencyProject)
280                         String extDocUrl = config.release ? calculateRemoteJavadocLink(pdep.group, pdep.name, pdep.version) : packageListLoc
281                         offlineLink(extDocUrl, packageListLoc)
282                     } else {
283                         String artifactName = "${dep.name}-${dep.version}".toString()
284                         if (!isExcluded(artifactName) && dep.name != 'unspecified' &&
285                             isNotBlank(dep.group) && isNotBlank(dep.version)) {
286                             links << calculateRemoteJavadocLink(dep.group, dep.name, dep.version)
287                         }
288                     }
289                 }
290             }
291 
292             links
293         }
294 
295         private boolean isExcluded(String artifactName) {
296             for (String s : excludes) {
297                 if (artifactName.matches(s)) {
298                     return true
299                 }
300             }
301             false
302         }

304         private String calculateRemoteJavadocLink(String group, String name, String version) {
305             String normalizedGroup = group.replace('.', '/')
306             "https://oss.sonatype.org/service/local/repositories/releases/archive/$normalizedGroup/$name/$version/$name-$version-javadoc.jar/!/".toString()
307         }

316         @CompileDynamic
317         private String calculateLocalJavadocLink(Project dependentProject, Project project) {
318             ProjectConfigurationExtension config = resolveEffectiveConfig(project) ?: resolveConfig(project)
319 
320             Task taskDependency = null
321             if (config.javadoc.enabled) {
322                 taskDependency = project.tasks.findByName('javadoc')
323             }
324 
325             if (config.groovydoc.enabled && config.groovydoc.replaceJavadoc) {
326                 taskDependency = project.tasks.findByName('groovydoc')
327             }
328 
329             dependentProject.tasks.findByName('javadoc').dependsOn(taskDependency)
330             taskDependency.destinationDir.absolutePath.replace('\\', '/')
331         }
332     }
