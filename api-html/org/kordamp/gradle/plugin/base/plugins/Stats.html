030 /**
031  * @author Andres Almiray
032  * @since 0.8.0

034 @CompileStatic
035 @Canonical
036 class Stats extends AbstractFeature {
037     static final String XML = 'xml'
038     static final String HTML = 'html'
039     static final String TXT = 'txt'
040 
041     Map<String, String> counters = [:]
042     Map<String, Map<String, String>> paths = [:]
043     List<String> formats = ['xml', 'txt']
044 
045     private final Set<Project> projects = new LinkedHashSet<>()
046     private final Set<TaskProvider<? extends Task>> statsTasks = new LinkedHashSet<>()
047 
048     Stats(Project project) {
049         super(project)
050         paths.putAll(defaultPaths())
051     }
052 
053     @Override
054     String toString() {
055         toMap().toString()
056     }
057 
058     @CompileDynamic
059     Map<String, Map<String, Object>> toMap() {
060         Map map = [enabled: enabled]
061 
062         if (enabled) {
063             map.formats = formats
064             map.counters = counters
065             map.paths = paths
066         }
067 
068         ['stats': map]
069     }
070 
071     static Map<String, Map<String, String>> defaultPaths() {
072         Map<String, Map<String, String>> basePaths = [:]
073 
074         [
075             java      : 'Java',
076             groovy    : 'Groovy',
077             scala     : 'Scala',
078             kt        : 'Kotlin',
079             js        : 'Javascript',
080             css       : 'CSS',
081             scss      : 'SASS',
082             xml       : 'XML',
083             html      : 'HTML',
084             fxml      : 'FXML',
085             properties: 'Properties',
086             sql       : 'SQL',
087             yaml      : 'YAML',
088             clojure   : 'Clojure'
089         ].each { extension, name ->
090             ['test', 'integration-test', 'functional-test'].each { source ->
091                 String classifier = getPropertyNameForLowerCaseHyphenSeparatedName(source)
092                 basePaths[classifier + extension.capitalize()] = [name: name + ' ' + getNaturalName(classifier) + ' Sources', path: 'src/' + source, extension: extension]
093             }
094         }
095 
096         basePaths.java = [name: 'Java Sources', path: '.*', extension: 'java']
097         basePaths.groovy = [name: 'Groovy Sources', path: '.*', extension: 'groovy']
098         basePaths.scala = [name: 'Scala Sources', path: '.*', extension: 'scala']
099         basePaths.kt = [name: 'Kotlin Sources', path: '.*', extension: 'kt']
100         basePaths.js = [name: 'Javascript Sources', path: '.*', extension: 'js']
101         basePaths.css = [name: 'CSS Sources', path: '.*', extension: 'css']
102         basePaths.scss = [name: 'SASS Sources', path: '.*', extension: 'scss']
103         basePaths.xml = [name: 'XML Sources', path: '.*', extension: 'xml']
104         basePaths.html = [name: 'HTML Sources', path: '.*', extension: 'html']
105         basePaths.fxml = [name: 'FXML Sources', path: '.*', extension: 'fxml']
106         basePaths.properties = [name: 'Properties', path: '.*', extension: 'properties']
107         basePaths.sql = [name: 'SQL', path: '.*', extension: 'sql']
108         basePaths.yaml = [name: 'Yaml', path: '.*', extension: 'yaml']
109         basePaths.clj = [name: 'Clojure', path: '.*', extension: 'clj']
110 
111         basePaths
112     }
113 
114     void copyInto(Stats copy) {
115         super.copyInto(copy)
116         copy.counters.putAll(counters)
117         copy.paths.putAll(paths)
118         List<String> fmts = new ArrayList<>(copy.formats)
119         copy.formats.clear()
120         copy.formats.addAll((fmts + formats).unique())
121         copy.formats.addAll(formats)
122     }
123 
124     static void merge(Stats o1, Stats o2) {
125         AbstractFeature.merge(o1, o2)
126         o1.counters.putAll(o2.counters)
127         o1.paths.putAll(o2.paths)
128         List<String> fmts = new ArrayList<>(o1.formats)
129         o1.formats.clear()
130         o1.formats.addAll((fmts + o2?.formats).unique())
131         o1.projects().addAll(o2.projects())
132         o1.statsTasks().addAll(o2.statsTasks())
133     }
134 
135     Set<Project> projects() {
136         projects
137     }
138 
139     Set<TaskProvider<? extends Task>> statsTasks() {
140         statsTasks
141     }
