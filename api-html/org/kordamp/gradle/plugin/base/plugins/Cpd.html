18 package org.kordamp.gradle.plugin.base.plugins
19 
20 import groovy.transform.CompileStatic
21 import org.gradle.api.Project
22 import org.kordamp.gradle.plugin.base.ProjectConfigurationExtension
23 import org.kordamp.gradle.plugin.base.internal.DefaultVersions
24 
25 /**
26  * @author Andres Almiray
27  * @since 0.37.0
28  */
29 @CompileStatic
30 class Cpd extends AbstractQualityFeature {
31     static final String PLUGIN_ID = 'org.kordamp.gradle.cpd'
32 
33     int minimumTokenCount
34     String encoding
35     String language
36     Boolean ignoreLiterals
37     Boolean ignoreIdentifiers
38     Boolean ignoreAnnotations
39 
40     Cpd(ProjectConfigurationExtension config, Project project) {
41         super(config, project, PLUGIN_ID, 'cpd')
42         toolVersion = DefaultVersions.INSTANCE.pmdVersion
43     }
44 
45     @Override
46     protected AbstractFeature getParentFeature() {
47         return project.rootProject.extensions.getByType(ProjectConfigurationExtension).quality.cpd
48     }
49 
50     @Override
51     void postMerge() {
52         minimumTokenCount = minimumTokenCount <= 0 ? 50 : minimumTokenCount
53         encoding = encoding ?: 'UTF-8'
54         language = language ?: 'java'
55         super.postMerge()
56     }
57 
58     @Override
59     protected void populateMapDescription(Map<String, Object> map) {
60         super.populateMapDescription(map)
61         map.language = this.language
62         map.encoding = this.encoding
63         map.minimumTokenCount = this.minimumTokenCount
64         map.ignoreLiterals = this.language
65         map.ignoreIdentifiers = this.language
66         map.ignoreAnnotations = this.language
67     }
68 
69     @Override
70     protected boolean hasBasePlugin(Project project) {
71         project.pluginManager.hasPlugin('java-base')
72     }
73 
74     boolean getIgnoreLiterals() {
75         null != ignoreLiterals && ignoreLiterals
76     }
77 
78     boolean getIgnoreIdentifiers() {
79         null != ignoreIdentifiers && ignoreIdentifiers
80     }
81 
82     boolean getIgnoreAnnotations() {
83         null != ignoreAnnotations && ignoreAnnotations
84     }
85 
86     static void merge(Cpd o1, Cpd o2) {
87         AbstractQualityFeature.merge(o1, o2)
88         o1.language = o1.language ?: o2.language
89         o1.encoding = o1.encoding ?: o2.encoding
90         o1.minimumTokenCount = o1.minimumTokenCount ?: o2.minimumTokenCount
91         o1.ignoreLiterals = o1.@ignoreLiterals != null ? o1.getIgnoreLiterals() : o2.getIgnoreLiterals()
92         o1.ignoreIdentifiers = o1.@ignoreIdentifiers != null ? o1.getIgnoreIdentifiers() : o2.getIgnoreIdentifiers()
93         o1.ignoreAnnotations = o1.@ignoreAnnotations != null ? o1.getIgnoreAnnotations() : o2.getIgnoreAnnotations()
94     }
95 }
