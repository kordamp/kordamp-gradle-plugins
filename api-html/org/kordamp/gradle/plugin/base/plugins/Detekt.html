031 @CompileStatic
032 @Canonical
033 class Detekt extends AbstractFeature {
034     static final String PLUGIN_ID = 'org.kordamp.gradle.detekt'
035     static final String KOTLIN_JVM_PLUGIN_ID = 'org.jetbrains.kotlin.jvm'
036 
037     File configFile
038     File baselineFile
039     boolean parallel = true
040     boolean buildUponDefaultConfig = false
041     boolean disableDefaultRuleSets = false
042     boolean ignoreFailures = true
043     String toolVersion = '1.2.2'
044     final Aggregate aggregate
045 
046     private boolean parallelSet
047     private boolean buildUponDefaultConfigSet
048     private boolean disableDefaultRuleSetsSet
049     private boolean ignoreFailuresSet
050 
051     Detekt(ProjectConfigurationExtension config, Project project) {
052         super(config, project)
053         aggregate = new Aggregate(config, project)
054         doSetEnabled(project.plugins.findPlugin(PLUGIN_ID) != null)
055     }
056 
057     @Override
058     String toString() {
059         toMap().toString()
060     }
061 
062     @Override
063     Map<String, Map<String, Object>> toMap() {
064         Map<String, Object> map = new LinkedHashMap<String, Object>(
065             enabled: enabled,
066             configFile: configFile,
067             baselineFile: baselineFile,
068             parallel: parallel,
069             buildUponDefaultConfig: buildUponDefaultConfig,
070             disableDefaultRuleSets: disableDefaultRuleSets,
071             ignoreFailures: ignoreFailures,
072             toolVersion: toolVersion
073         )
074         if (isRoot()) {
075             map.putAll(aggregate.toMap())
076         }
077         new LinkedHashMap<>('detekt': map)
078     }
079 
080     void normalize() {
081         if (null == configFile) {
082             File file = project.rootProject.file("config/detekt/${project.name}.yml")
083             if (!file.exists()) {
084                 file = project.rootProject.file('config/detekt/detekt.yml')
085             }
086             configFile = file
087         }
088 
089         if (!enabledSet) {
090             if (isRoot()) {
091                 if (project.childProjects.isEmpty()) {
092                     setEnabled(project.pluginManager.hasPlugin(KOTLIN_JVM_PLUGIN_ID) && isApplied())
093                 } else {
094                     setEnabled(project.childProjects.values().any { p -> p.pluginManager.hasPlugin(KOTLIN_JVM_PLUGIN_ID) && isApplied()})
095                 }
096             } else {
097                 setEnabled(project.pluginManager.hasPlugin(KOTLIN_JVM_PLUGIN_ID) && isApplied())
098             }
099         }
100     }

146     void copyInto(Detekt copy) {
147         super.copyInto(copy)
148         copy.@parallel = parallel
149         copy.@parallelSet = parallelSet
150         copy.@buildUponDefaultConfig = buildUponDefaultConfig
151         copy.@buildUponDefaultConfigSet = buildUponDefaultConfigSet
152         copy.@disableDefaultRuleSets = disableDefaultRuleSets
153         copy.@disableDefaultRuleSetsSet = disableDefaultRuleSetsSet
154         copy.@ignoreFailures = ignoreFailures
155         copy.@ignoreFailuresSet = ignoreFailuresSet
156         copy.configFile = configFile
157         copy.baselineFile = baselineFile
158         copy.toolVersion = toolVersion
159         aggregate.copyInto(copy.aggregate)
160     }

162     static void merge(Detekt o1, Detekt o2) {
163         AbstractFeature.merge(o1, o2)
164         o1.setParallel((boolean) (o1.parallelSet ? o1.parallel : o2.parallel))
165         o1.setBuildUponDefaultConfig((boolean) (o1.buildUponDefaultConfigSet ? o1.buildUponDefaultConfig : o2.buildUponDefaultConfig))
166         o1.setDisableDefaultRuleSets((boolean) (o1.disableDefaultRuleSetsSet ? o1.disableDefaultRuleSets : o2.disableDefaultRuleSets))
167         o1.setIgnoreFailures((boolean) (o1.ignoreFailuresSet ? o1.ignoreFailures : o2.ignoreFailures))
168         o1.configFile = o1.configFile ?: o2.configFile
169         o1.baselineFile = o1.baselineFile ?: o2.baselineFile
170         o1.toolVersion = o1.toolVersion ?: o2.toolVersion
171         o1.aggregate.merge(o2.aggregate)
172     }

174     @CompileStatic
175     static class Aggregate {
176         Boolean enabled
177         private final Set<Project> excludedProjects = new LinkedHashSet<>()
178 
179         private final ProjectConfigurationExtension config
180         private final Project project
181 
182         Aggregate(ProjectConfigurationExtension config, Project project) {
183             this.config = config
184             this.project = project
185         }
186 
187         Map<String, Object> toMap() {
188             Map<String, Object> map = new LinkedHashMap<String, Object>()
189 
190             map.enabled = getEnabled()
191             map.excludedProjects = excludedProjects
192 
193             new LinkedHashMap<>('aggregate': map)
194         }
195 
196         boolean getEnabled() {
197             this.@enabled == null || this.@enabled
198         }
199 
200         void copyInto(Aggregate copy) {
201             copy.@enabled = this.@enabled
202             copy.excludedProjects.addAll(excludedProjects)
203         }
204 
205         Aggregate copyOf() {
206             Aggregate copy = new Aggregate(config, project)
207             copyInto(copy)
208             copy
209         }
210 
211         Aggregate merge(Aggregate other) {
212             Aggregate copy = copyOf()
213             copy.enabled = copy.@enabled != null ? copy.getEnabled() : other.getEnabled()
214             copy
215         }
216 
217         Set<Project> excludedProjects() {
218             excludedProjects
219         }
220     }
