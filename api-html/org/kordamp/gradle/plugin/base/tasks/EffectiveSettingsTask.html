18 package org.kordamp.gradle.plugin.base.tasks
19 
20 import groovy.transform.CompileStatic
21 import org.gradle.api.tasks.TaskAction
22 import org.gradle.api.tasks.options.Option
23 
24 import static org.kordamp.gradle.PluginUtils.resolveEffectiveConfig
25 import static org.kordamp.gradle.StringUtils.isNotBlank
26 
27 /**
28  * @author Andres Almiray
29  * @since 0.8.0
30  */
31 @CompileStatic
32 class EffectiveSettingsTask extends AbstractReportingTask {
33     private final List<String> COVERAGE = ['coveralls', 'jacoco'].asImmutable()
34     private final List<String> DOCS = ['guide', 'groovydoc', 'javadoc', 'kotlindoc', 'scaladoc', 'sourceHtml', 'sourceXref'].asImmutable()
35     private final List<String> QUALITY = ['checkstyle', 'codenarc', 'detekt', 'pmd', 'sonar', 'spotbugs'].asImmutable()
36 
37     private String section
38     private Set<String> sections
39 
40     @Option(option = 'section', description = 'The section to generate the report for.')
41     void setSection(String section) {
42         this.section = section
43     }
44 
45     @Option(option = 'sections', description = 'The sections to generate the report for.')
46     void setSections(String sections) {
47         if (isNotBlank(sections)) {
48             this.sections = (sections.split(',').collect { it.trim() }) as Set
49         }
50     }
51 
52     @TaskAction
53     void displayEffectiveSettings() {
54         Map<String, Object> map = resolveEffectiveConfig(project).toMap()
55 
56         if (sections) {
57             sections.each { s ->
58                 printSection(map, s)
59             }
60         } else if (section) {
61             printSection(map, section)
62         } else {
63             doPrint(map, 0)
64         }
65     }
66 
67     private void printSection(Map<String, Object> map, String section) {
68         if (map.containsKey(section)) {
69             println "${section}:"
70             doPrint((Map<String, ?>) map[section], 1)
71         } else if (section in QUALITY) {
72             printSection((Map<String, Object>) map.quality, section)
73         } else if (section in DOCS) {
74             printSection((Map<String, Object>) map.docs, section)
75         } else if (section in COVERAGE) {
76             printSection((Map<String, Object>) map.coverage, section)
77         } else {
78             throw new IllegalStateException("Unknown section '$section'")
79         }
80     }
81 }
