018 package org.kordamp.gradle.plugin.base.tasks
019 
020 import groovy.transform.CompileStatic
021 import org.gradle.api.artifacts.Configuration
022 import org.gradle.api.file.FileCollection
023 import org.gradle.api.provider.Property
024 import org.gradle.api.tasks.Input
025 import org.gradle.api.tasks.Optional
026 import org.gradle.api.tasks.TaskAction
027 import org.gradle.api.tasks.options.Option
028 
029 import static org.kordamp.gradle.StringUtils.isNotBlank
030 
031 /**
032  * @author Andres Almiray
033  * @since 0.24.0
034  */
035 @CompileStatic
036 class ConfigurationSettingsTask extends AbstractReportingTask {
037     @Input @Optional String configuration
038     @Input @Optional Set<String> configurations
039 
040     private final Property<Boolean> showPaths = project.objects.property(Boolean)
041 
042     @Option(option = 'show-paths', description = 'Display path information (OPTIONAL).')
043     void setShowPaths(boolean showPaths) {
044         this.showPaths.set(showPaths)
045     }
046 
047     @Input
048     boolean isShowPaths() {
049         showPaths.getOrElse(false)
050     }
051 
052     @Option(option = 'configuration', description = 'The configuration to generate the report for.')
053     void setConfiguration(String configuration) {
054         this.configuration = configuration
055     }
056 
057     @Option(option = 'configurations', description = 'The configurations to generate the report for.')
058     void setConfigurations(String configurations) {
059         if (isNotBlank(configurations)) {
060             this.configurations = (configurations.split(',').collect { it.trim() }) as Set
061         }
062     }
063 
064     @TaskAction
065     void report() {
066         if (configurations) {
067             configurations.each { c ->
068                 printConfiguration((Configuration) project.configurations.findByName(c))
069             }
070         } else if (configuration) {
071             try {
072                 printConfiguration((Configuration) project.configurations.findByName(configuration))
073             } catch (NullPointerException e) {
074                 throw new IllegalStateException("No matching '${this.configuration}' configuration was found")
075             }
076         } else {
077             project.configurations.each { c ->
078                 printConfiguration(c)
079             }
080         }
081     }
082 
083     private void printConfiguration(Configuration configuration) {
084         print(configuration.name + ':', 0)
085         configuration.properties.sort().each { name, value ->
086             if (value instanceof FileCollection ||
087                 (name in ['class', 'name', 'all', 'asDynamicObject', 'asFileTree', 'asFileTrees', 'asPath', 'convention', 'conventionMapping', 'buildDependencies',
088                           'dependenciesResolver', 'dependencyResolutionListeners', 'extensions', 'outgoing', 'module', 'resolutionStrategy', 'resolvedConfiguration',
089                           'allArtifacts', 'allDependencies', 'artifacts'])) {
090                 return
091             }
092             if (value instanceof Collection) {
093                 doPrintCollection(name.toString(), (Collection) value, 1)
094             } else {
095                 doPrintMapEntry(name.toString(), value, 1)
096             }
097         }
098         if (isShowPaths() && configuration.canBeResolved) {
099             configuration.properties.sort().each { name, value ->
100                 if (value instanceof FileCollection) {
101                     doPrintCollection(name.toString(), value, 1)
102                 }
103             }
104         }
105 
106         println ' '
107     }
108 }
