18 package org.kordamp.gradle.plugin.base.tasks

20 
21 import groovy.transform.CompileStatic
22 import org.gradle.api.Plugin
23 import org.gradle.api.tasks.TaskAction
24 import org.kordamp.gradle.plugin.KordampPlugin
25 
26 import java.util.jar.JarEntry
27 import java.util.jar.JarFile
28 import java.util.regex.Matcher
29 
30 /**
31  * @author Andres Almiray
32  * @since 0.11.0
33  */
34 @CompileStatic
35 class PluginsTask extends AbstractReportingTask {
36     @TaskAction
37     void report() {
38         Map<String, Map<String, Object>> plugins = new LinkedHashMap<String, Map<String, Object>>()
39 
40         Map<String, String> pluginMetadata = new LinkedHashMap<String, String>()
41         Enumeration<URL> e = PluginsTask.classLoader.getResources('META-INF/gradle-plugins')
42         while (e.hasMoreElements()) {
43             extractMetadata(e.nextElement(), pluginMetadata)
44         }
45         e = org.gradle.api.plugins.BasePlugin.classLoader.getResources('META-INF/gradle-plugins')
46         while (e.hasMoreElements()) {
47             extractMetadata(e.nextElement(), pluginMetadata)
48         }
49 
50         project.plugins.eachWithIndex { plugin, index -> plugins.putAll(PluginsTask.doReport(plugin, index, pluginMetadata)) }
51 
52         doPrint(plugins, 0)
53     }
54 
55     private void extractMetadata(URL url, Map<String, String> pluginMetadata) {
56         if (url.protocol != 'jar') return
57 
58         JarFile jarFile = new JarFile(url.toString()[9..url.toString().indexOf('!') - 1])
59         Enumeration<JarEntry> entries = jarFile.entries()
60         while (entries.hasMoreElements()) {
61             JarEntry entry = entries.nextElement()
62             Matcher matcher = (entry.name =~ /META-INF\/gradle-plugins\/(.+)\.properties/)
63             if (matcher.matches()) {
64                 Properties props = new Properties()
65                 props.load(jarFile.getInputStream(entry))
66                 pluginMetadata.put((String) props.'implementation-class', matcher.group(1))
67             }
68         }
69     }
70 
71     private static Map<String, Map<String, Object>> doReport(Plugin plugin, int index, Map<String, String> pluginMetadata) {
72         Map<String, Object> map = new LinkedHashMap<>()
73 
74         map.id = (pluginMetadata[plugin.class.name] ?: plugin.class.name) - 'org.gradle.'
75         map.implementationClass = plugin.class.name
76         if (plugin instanceof KordampPlugin) {
77             map.enabled = plugin.enabled
78         }
79 
80         new LinkedHashMap<>([('plugin ' + index): map])
81     }
82 }
