18 package org.kordamp.gradle.plugin.base.tasks
19 
20 import groovy.transform.CompileStatic
21 import org.gradle.api.tasks.TaskAction
22 import org.gradle.api.tasks.compile.GroovyCompile
23 
24 /**
25  * @author Andres Almiray
26  * @since 0.20.0
27  */
28 @CompileStatic
29 class GroovyCompilerSettingsTask extends AbstractSettingsTask {
30     @TaskAction
31     void report() {
32         if (tasks) {
33             tasks.each { t ->
34                 printTask((GroovyCompile) project.tasks.findByName(t))
35             }
36         } else if (task) {
37             printTask((GroovyCompile) project.tasks.findByName(task))
38         } else {
39             project.tasks.withType(GroovyCompile).each { t ->
40                 printTask(t)
41             }
42         }
43     }
44 
45     private void printTask(GroovyCompile task) {
46         print(task.name + ':', 0)
47         doPrintCollection('includes', task.includes, 1)
48         doPrintCollection('excludes', task.excludes, 1)
49         doPrintMapEntry('sourceCompatibility', task.sourceCompatibility, 1)
50         doPrintMapEntry('targetCompatibility', task.targetCompatibility, 1)
51         print('groovyOptions:', 1)
52         doPrintMapEntry('configurationScript', task.groovyOptions.configurationScript, 2)
53         doPrintMapEntry('encoding', task.groovyOptions.encoding, 2)
54         doPrintMapEntry('failOnError', task.groovyOptions.failOnError, 2)
55         doPrintCollection('fileExtensions', task.groovyOptions.fileExtensions, 2)
56         doPrintMapEntry('fork', task.groovyOptions.fork, 2)
57         print('forkOptions:', 2)
58         doPrintCollection('jvmArgs', task.options.forkOptions.jvmArgs, 3)
59         doPrintMapEntry('memoryInitialSize', task.options.forkOptions.memoryInitialSize, 3)
60         doPrintMapEntry('memoryMaximumSize', task.options.forkOptions.memoryMaximumSize, 3)
61         doPrintMapEntry('javaAnnotationProcessing', task.groovyOptions.javaAnnotationProcessing, 2)
62         doPrintMapEntry('keepStubs', task.groovyOptions.keepStubs, 2)
63         doPrintMapEntry('listFiles', task.groovyOptions.listFiles, 2)
64         doPrintMap('optimizationOptions', task.groovyOptions.optimizationOptions, 2)
65         doPrintMapEntry('stubDir', task.groovyOptions.stubDir, 2)
66         doPrintMapEntry('verbose', task.groovyOptions.verbose, 2)
67         print('options:', 1)
68         doPrintCollection('compilerArgs', task.options.compilerArgs, 2)
69         doPrintMapEntry('debug', task.options.debug, 2)
70         print('debugOptions:', 2)
71         doPrintMapEntry('debugLevel', task.options.debugOptions.debugLevel, 3)
72         doPrintMapEntry('deprecation', task.options.deprecation, 2)
73         doPrintMapEntry('encoding', task.options.encoding, 2)
74         doPrintMapEntry('extensionDirs', task.options.extensionDirs, 2)
75         doPrintMapEntry('failOnError', task.options.failOnError, 2)
76         doPrintMapEntry('fork', task.options.fork, 2)
77         print('forkOptions:', 2)
78         doPrintMapEntry('executable', task.options.forkOptions.executable, 3)
79         doPrintMapEntry('javaHome', task.options.forkOptions.javaHome, 3)
80         doPrintCollection('jvmArgs', task.options.forkOptions.jvmArgs, 3)
81         doPrintMapEntry('memoryInitialSize', task.options.forkOptions.memoryInitialSize, 3)
82         doPrintMapEntry('memoryMaximumSize', task.options.forkOptions.memoryMaximumSize, 3)
83         doPrintMapEntry('incremental', task.options.incremental, 2)
84         doPrintMapEntry('listFiles', task.options.listFiles, 2)
85         doPrintMapEntry('verbose', task.options.verbose, 2)
86         doPrintMapEntry('warnings', task.options.warnings, 2)
87         if (isShowPaths()) {
88             doPrintCollection('annotationProcessorPath', task.options.annotationProcessorPath, 1)
89             doPrintCollection('bootstrapClasspath', task.options.bootstrapClasspath, 1)
90             doPrintCollection('classpath', task.classpath, 1)
91             doPrintCollection('sourcepath', task.options.sourcepath, 1)
92         }
93         println ' '
94     }
95 }
