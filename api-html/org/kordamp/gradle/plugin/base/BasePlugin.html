325     @Named('base')
326     private class BaseProjectEvaluatedListener implements ProjectEvaluatedListener {
327         @Override
328         void projectEvaluated(Project project) {
329             ProjectConfigurationExtension rootExtension = project.rootProject.extensions.findByType(ProjectConfigurationExtension)
330             ProjectConfigurationExtension extension = project.extensions.findByType(ProjectConfigurationExtension)
331             extension.normalize()
332 
333             boolean validate = PluginUtils.checkFlag(ORG_KORDAMP_GRADLE_BASE_VALIDATE, true)
334 
335             List<String> errors = []
336             if (isRootProject(project)) {
337                 // extension == rootExtension
338                 extension.postMerge()
339             } else {
340                 // parent project may not have applied kordamp.base
341                 if (rootExtension) {
342                     extension.merge(rootExtension)
343                 } else {
344                     extension.postMerge()
345                 }
346             }
347             if (validate) errors.addAll(extension.validate())
348 
349             if (validate && errors) {
350                 errors.each { project.logger.error(it) }
351                 boolean ideaSync = (System.getProperty('idea.sync.active') ?: 'false').toBoolean()
352                 if (!ideaSync) throw new GradleException("Project ${project.name} has not been properly configured.")
353             }
354         }
355     }
