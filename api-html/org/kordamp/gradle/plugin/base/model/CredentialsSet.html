018 package org.kordamp.gradle.plugin.base.model
019 
020 import groovy.transform.CompileStatic
021 import org.gradle.api.Action
022 import org.kordamp.gradle.util.ConfigureUtil
023 
024 /**
025  * @author Andres Almiray
026  * @since 0.8.0
027  */
028 @CompileStatic
029 class CredentialsSet {
030     static final String GITHUB = 'github'
031     static final String SONATYPE = 'sonatype'
032 
033     final Map<String, Credentials> credentialsMap = new LinkedHashMap<>()
034 
035     Map<String, Map<String, Object>> toMap() {
036         if (isEmpty()) return [:]
037 
038         (Map<String, Map<String, Object>>) credentialsMap.collectEntries { k, v ->
039             [(k): v.toMap()]
040         }
041     }
042 
043     void github(Action<? super Credentials> action) {
044         action.execute(credentialsMap.computeIfAbsent(GITHUB, { k -> new Credentials() }))
045     }
046 
047     void sonatype(Action<? super Credentials> action) {
048         action.execute(credentialsMap.computeIfAbsent(SONATYPE, { k -> new Credentials() }))
049     }
050 
051     void github(@DelegatesTo(strategy = Closure.DELEGATE_FIRST, value = Credentials) Closure<Void> action) {
052         ConfigureUtil.configure(action, credentialsMap.computeIfAbsent(GITHUB, { k -> new Credentials() }))
053     }
054 
055     void sonatype(@DelegatesTo(strategy = Closure.DELEGATE_FIRST, value = Credentials) Closure<Void> action) {
056         ConfigureUtil.configure(action, credentialsMap.computeIfAbsent(SONATYPE, { k -> new Credentials() }))
057     }
058 
059     void named(Action<? super Credentials> action) {
060         Credentials c = new Credentials()
061         action.execute(c)
062         credentialsMap.put(c.name, Credentials.merge(c, getCredentials(c.name)))
063     }
064 
065     void named(@DelegatesTo(strategy = Closure.DELEGATE_FIRST, value = Credentials) Closure<Void> action) {
066         Credentials c = new Credentials()
067         ConfigureUtil.configure(action, c)
068         credentialsMap.put(c.name, Credentials.merge(c, getCredentials(c.name)))
069     }
070 
071     Credentials getGithub() {
072         credentialsMap.get(GITHUB)
073     }
074 
075     Credentials getSonatype() {
076         credentialsMap.get(SONATYPE)
077     }
078 
079     Credentials getCredentials(String name) {
080         credentialsMap.get(name)
081     }
082 
083     void copyInto(CredentialsSet credentialsSet) {
084         credentialsMap.each { k, v -> credentialsSet.credentialsMap.put(k, v) }
085     }
086 
087     static void merge(CredentialsSet o1, CredentialsSet o2) {
088         if (!o1.credentialsMap) {
089             if (o2?.credentialsMap) {
090                 o1.credentialsMap.putAll(o2.credentialsMap)
091             }
092         } else {
093             o1.credentialsMap.each { k, v ->
094                 Credentials merged = Credentials.merge(v, o2?.credentialsMap?.get(k))
095                 o1.credentialsMap.put(k, merged)
096             }
097             o2?.credentialsMap.each { k, v ->
098                 if (!o1.credentialsMap.containsKey(k)) {
099                     o1.credentialsMap.put(k, v)
100                 }
101             }
102         }
103     }
104 
105     boolean isEmpty() {
106         credentialsMap.isEmpty()
107     }
108 }
