18 package org.kordamp.gradle.plugin.base.model
19 
20 import groovy.transform.Canonical
21 import groovy.transform.CompileStatic
22 import groovy.transform.ToString
23 import org.gradle.api.Project
24 
25 import static org.kordamp.gradle.StringUtils.isBlank
26 
27 /**
28  * @author Andres Almiray
29  * @since 0.9.0
30  */
31 @CompileStatic
32 @Canonical
33 @ToString(includeNames = true)
34 class Scm {
35     Boolean enabled
36     String url
37     String tag
38     String connection
39     String developerConnection
40 
41     @Override
42     String toString() {
43         toMap().toString()
44     }
45 
46     Map<String, Object> toMap() {
47         new LinkedHashMap<String, Object>([
48             enabled            : getEnabled(),
49             url                : url,
50             tag                : tag,
51             connection         : connection,
52             developerConnection: developerConnection
53         ])
54     }
55 
56     boolean getEnabled() {
57         this.@enabled == null || this.@enabled
58     }
59 
60     void copyInto(Scm copy) {
61         copy.enabled = this.@enabled
62         copy.url = url
63         copy.tag = tag
64         copy.connection = connection
65         copy.developerConnection = developerConnection
66     }
67 
68     static void merge(Scm o1, Scm o2) {
69         o1.enabled = o1.@enabled != null ? o1.getEnabled() : o2.getEnabled()
70         o1.url = o1.url ?: o2.url
71         o1.tag = o1.tag ?: o2.tag
72         o1.connection = o1.connection ?: o2.connection
73         o1.developerConnection = o1.developerConnection ?: o2.developerConnection
74     }
75 
76     List<String> validate(Project project) {
77         List<String> errors = []
78 
79         if (getEnabled() && isBlank(url)) {
80             errors << "[${project.name}] Project links:url is blank".toString()
81         }
82 
83         errors
84     }
85 
86     boolean isEmpty() {
87         !getEnabled() && isBlank(url)
88     }
89 }
