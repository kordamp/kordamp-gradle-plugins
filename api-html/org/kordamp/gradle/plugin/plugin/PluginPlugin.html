047 /**
048  * @author Andres Almiray
049  * @since 0.16.0
050  */
051 @CompileStatic
052 class PluginPlugin extends AbstractKordampPlugin {
053     Project project
054 
055     PluginPlugin() {
056         super(Plugins.PLUGIN_ID)
057     }
058 
059     void apply(Project project) {
060         this.project = project
061 
062         configureProject(project)
063         project.childProjects.values().each {
064             it.pluginManager.apply(PluginPlugin)
065         }
066     }
067 
068     private void configureProject(Project project) {
069         if (hasBeenVisited(project)) {
070             return
071         }
072         setVisited(project, true)
073 
074         BasePlugin.applyIfMissing(project)
075 
076         project.tasks.register('listPluginDescriptors', ListPluginDescriptorsTask.class,
077             new Action<ListPluginDescriptorsTask>() {
078                 void execute(ListPluginDescriptorsTask t) {
079                     t.group = 'Plugin development'
080                     t.description = 'Lists plugin descriptors from plugin declarations.'
081                     t.declarations.set(project.extensions.findByType(GradlePluginDevelopmentExtension).plugins)
082                 }
083             })
084 
085         project.tasks.register('publishRelease', DefaultTask,
086             new Action<DefaultTask>() {
087                 @Override
088                 void execute(DefaultTask t) {
089                     t.group = 'Publishing'
090                     t.description = 'Publishes plugin artifacts to Bintray and the Gradle Plugin Portal'
091                     Task bintrayUpload = project.tasks.findByName('bintrayUpload')
092                     if (bintrayUpload) {
093                         t.dependsOn(bintrayUpload)
094                     }
095                     t.dependsOn(project.tasks.named('publishPlugins'))
096                 }
097             })
098 
099         addProjectEvaluatedListener(project, new PublishingProjectEvaluatedListener())
100 
101         project.pluginManager.apply(JavaGradlePluginPlugin)
102         project.pluginManager.apply(PublishPlugin)
103 
104         // Must execute after JavaGradlePluginPlugin|PublishPlugin
105         // because ${plugin.name}PluginMarkerMaven is created explicitly and not with `maybeCreate`.
106         project.afterEvaluate {
107             ProjectConfigurationExtension config = resolveConfig(project)
108             config.plugins.plugins.values().each { plugin ->
109                 String pluginName = plugin.name
110                 updatePluginPublication(project, config, pluginName)
111             }
112             updatePublications(project, config)
113         }
114     }
115 
116     @Named('plugin')
117     //@DependsOn(['publishing'])
118     private class PublishingProjectEvaluatedListener implements ProjectEvaluatedListener {
119         @Override
120         void projectEvaluated(Project project) {
121             ProjectConfigurationExtension config = resolveConfig(project)
122 
123             GradlePluginDevelopmentExtension gpdExt = project.extensions.findByType(GradlePluginDevelopmentExtension)
124             PluginBundleExtension pbExt = project.extensions.findByType(PluginBundleExtension)
125 
126             if (isBlank(pbExt.website) && isNotBlank(config.info.url)) {
127                 pbExt.website = config.info.url
128             }
129             if (isBlank(pbExt.vcsUrl)) {
130                 pbExt.vcsUrl = config.info.resolveScmLink()
131             }
132             if (isBlank(pbExt.description)) {
133                 pbExt.description = config.info.description
134             }
135             if (!pbExt.tags) {
136                 pbExt.tags = config.info.tags
137             }
138 
139             if (isBlank(pbExt.mavenCoordinates.groupId)) {
140                 pbExt.mavenCoordinates.groupId = project.group
141             }
142             if (isBlank(pbExt.mavenCoordinates.artifactId)) {
143                 pbExt.mavenCoordinates.artifactId = project.name
144             }
145 
146             if (isBlank(pbExt.mavenCoordinates.version)) {
147                 pbExt.mavenCoordinates.version = project.version
148             }
149 
150             config.plugins.plugins.values().each { plugin ->
151                 String pluginName = plugin.name
152 
153                 PluginDeclaration pd = gpdExt.plugins.maybeCreate(pluginName)
154                 pd.id = plugin.id
155                 pd.displayName = plugin.displayName ?: config.info.description
156                 pd.description = plugin.description ?: config.info.description
157                 pd.implementationClass = plugin.implementationClass
158 
159                 PluginConfig pc = pbExt.plugins.maybeCreate(pluginName)
160                 pc.id = plugin.id
161                 pc.displayName = plugin.displayName ?: config.info.description
162                 pc.description = plugin.description ?: config.info.description
163                 pc.tags = plugin.resolveTags(config)
164                 pc.version = project.version
165             }
166         }
167     }
168 
169     @CompileDynamic
170     private void updatePluginPublication(Project project, ProjectConfigurationExtension config, String pluginName) {
171         PublishingUtils.configureSigning(config, project, pluginName + 'PluginMarkerMaven')
172     }
173 
174     @CompileDynamic
175     private void updatePublications(Project project, ProjectConfigurationExtension config) {
176         project.publishing {
177             publications {
178                 pluginMaven(MavenPublication) {
179                     PublishingUtils.configurePom(pom, config, config.publishing.pom)
180                 }
181             }
182         }
183 
184         PublishingUtils.configureSigning(config, project, 'pluginMaven')
185     }
