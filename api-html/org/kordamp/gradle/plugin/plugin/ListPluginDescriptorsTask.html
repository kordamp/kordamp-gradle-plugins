18 package org.kordamp.gradle.plugin.plugin
19 
20 import groovy.transform.CompileStatic
21 import org.gradle.api.provider.ListProperty
22 import org.gradle.api.tasks.Input
23 import org.gradle.api.tasks.TaskAction
24 import org.gradle.plugin.devel.PluginDeclaration
25 import org.kordamp.gradle.plugin.base.tasks.AbstractReportingTask
26 
27 /**
28  * @author Andres Almiray
29  * @since 0.16.0
30  */
31 @CompileStatic
32 class ListPluginDescriptorsTask extends AbstractReportingTask {
33     @Input
34     final ListProperty<PluginDeclaration> declarations
35 
36     ListPluginDescriptorsTask() {
37         this.declarations = project.objects.listProperty(PluginDeclaration)
38     }
39 
40     @TaskAction
41     void listPluginDescriptors() {
42         Map<String, Map<String, String>> map = [:]
43         for (PluginDeclaration declaration : declarations.get()) {
44             map.put(declaration.name, [
45                 id                 : declaration.id,
46                 implementationClass: declaration.implementationClass,
47                 displayName        : declaration.displayName,
48                 description        : declaration.description
49             ])
50         }
51 
52         doPrint(map, 0)
53     }
54 }
