042 /**
043  * Configures {@code aggregateApidocs} {@code aggregateApidocsJar} tasks on the root project.
044  *
045  * @author Andres Almiray
046  * @since 0.1.0
047  */
048 class ApidocPlugin extends AbstractKordampPlugin {
049     static final String AGGREGATE_APIDOCS_TASK_NAME = 'aggregateApidocs'
050     static final String AGGREGATE_JAVADOCS_TASK_NAME = 'aggregateJavadocs'
051     static final String AGGREGATE_JAVADOCS_JAR_TASK_NAME = 'aggregateJavadocsJar'
052     static final String AGGREGATE_GROOVYDOCS_TASK_NAME = 'aggregateGroovydocs'
053     static final String AGGREGATE_GROOVYDOCS_JAR_TASK_NAME = 'aggregateGroovydocsJar'
054 
055     Project project
056 
057     @CompileStatic
058     void apply(Project project) {
059         this.project = project
060 
061         if (isRootProject(project)) {
062             if (project.childProjects.size()) {
063                 project.childProjects.values().each {
064                     configureProject(it)
065                 }
066                 configureRootProject(project, true)
067             } else {
068                 configureProject(project)
069                 configureRootProject(project, false)
070             }
071         } else {
072             configureProject(project)
073         }
074     }
075 
076     @CompileStatic
077     static void applyIfMissing(Project project) {
078         if (!project.plugins.findPlugin(ApidocPlugin)) {
079             project.pluginManager.apply(ApidocPlugin)
080         }
081     }
082 
083     @CompileStatic
084     private void configureProject(Project project) {
085         if (hasBeenVisited(project)) {
086             return
087         }
088         setVisited(project, true)
089 
090         BasePlugin.applyIfMissing(project)
091         JavadocPlugin.applyIfMissing(project)
092         GroovydocPlugin.applyIfMissing(project)
093     }
094 
095     @CompileStatic
096     private void configureRootProject(Project project, boolean checkIfApplied) {
097         if (checkIfApplied && hasBeenVisited(project)) {
098             return
099         }
100         setVisited(project, true)
101 
102         BasePlugin.applyIfMissing(project)
103 
104         if (isRootProject(project) && !project.childProjects.isEmpty()) {
105             createAggregateGroovydocsTask(project, createAggregateJavadocsTask(project))
106             createAggregateApidocTask(project)
107 
108             project.gradle.addBuildListener(new BuildAdapter() {
109                 @Override
110                 void projectsEvaluated(Gradle gradle) {
111                     doConfigureRootProject(project)
112                 }
113             })
114         }
115     }
116 
117     private void doConfigureRootProject(Project project) {
118         ProjectConfigurationExtension effectiveConfig = resolveEffectiveConfig(project)
119         setEnabled(effectiveConfig.apidoc.enabled)
120 
121         if (!enabled) {
122             return
123         }
124 
125         if (!project.childProjects.isEmpty()) {
126             List<Javadoc> javadocs = []
127             project.tasks.withType(Javadoc) { Javadoc javadoc -> if (javadoc.name != AGGREGATE_JAVADOCS_TASK_NAME && javadoc.enabled) javadocs << javadoc }
128             project.childProjects.values().each { Project p ->
129                 if (p in effectiveConfig.apidoc.excludedProjects()) return
130                 p.tasks.withType(Javadoc) { Javadoc javadoc -> if (javadoc.enabled) javadocs << javadoc }
131             }
132             javadocs = javadocs.unique()
133 
134             List<Groovydoc> groovydocs = []
135             project.tasks.withType(Groovydoc) { Groovydoc groovydoc -> if (groovydoc.name != AGGREGATE_GROOVYDOCS_TASK_NAME && groovydoc.enabled) groovydocs << groovydoc }
136             project.childProjects.values().each { Project p ->
137                 if (p in effectiveConfig.apidoc.excludedProjects()) return
138                 p.tasks.withType(Groovydoc) { Groovydoc groovydoc -> if (groovydoc.enabled) groovydocs << groovydoc }
139             }
140             groovydocs = groovydocs.unique()
141 
142             Javadoc aggregateJavadocs = (Javadoc) project.tasks.findByName(AGGREGATE_JAVADOCS_TASK_NAME)
143             Jar aggregateJavadocsJar = (Jar) project.tasks.findByName(AGGREGATE_JAVADOCS_JAR_TASK_NAME)
144             Groovydoc aggregateGroovydocs = (Groovydoc) project.tasks.findByName(AGGREGATE_GROOVYDOCS_TASK_NAME)
145             Jar aggregateGroovydocsJar = (Jar) project.tasks.findByName(AGGREGATE_GROOVYDOCS_JAR_TASK_NAME)
146             Task aggregateApidocTask = project.tasks.findByName(AGGREGATE_APIDOCS_TASK_NAME)
147 
148             if (javadocs && !effectiveConfig.groovydoc.replaceJavadoc) {
149                 aggregateJavadocs.configure { task ->
150                     task.enabled = true
151                     task.dependsOn javadocs
152                     task.source javadocs.source
153                     task.classpath = project.files(javadocs.classpath)
154 
155                     effectiveConfig.javadoc.applyTo(task)
156                     task.options.footer = "Copyright &copy; ${effectiveConfig.info.copyrightYear} ${effectiveConfig.info.authors.join(', ')}. All rights reserved."
157                 }
158                 aggregateJavadocsJar.configure {
159                     enabled = true
160                     from aggregateJavadocs.destinationDir
161                 }
162 
163                 aggregateApidocTask.enabled = true
164                 aggregateApidocTask.dependsOn aggregateJavadocs
165             }
166 
167             if (groovydocs && effectiveConfig.groovydoc.enabled) {
168                 aggregateGroovydocs.configure { task ->
169                     task.enabled = true
170                     task.dependsOn groovydocs + javadocs
171                     task.source groovydocs.source + javadocs.source
172                     task.classpath = project.files(groovydocs.classpath + javadocs.classpath)
173                     task.groovyClasspath = project.files(groovydocs.groovyClasspath.flatten().unique())
174 
175                     effectiveConfig.groovydoc.applyTo(task)
176                     task.footer = "Copyright &copy; ${effectiveConfig.info.copyrightYear} ${effectiveConfig.info.authors.join(', ')}. All rights reserved."
177                 }
178                 aggregateGroovydocsJar.configure {
179                     enabled = true
180                     from aggregateGroovydocs.destinationDir
181                     archiveClassifier.set(effectiveConfig.groovydoc.replaceJavadoc ? 'javadoc' : 'groovydoc')
182                 }
183 
184                 aggregateApidocTask.enabled = true
185                 aggregateApidocTask.dependsOn aggregateGroovydocs
186             }
187         }
188     }

190     private TaskProvider<Javadoc> createAggregateJavadocsTask(Project project) {
191         TaskProvider<Javadoc> aggregateJavadocs = project.tasks.register(AGGREGATE_JAVADOCS_TASK_NAME, Javadoc,
192                 new Action<Javadoc>() {
193                     @Override
194                     void execute(Javadoc t) {
195                         t.enabled = false
196                         t.group = JavaBasePlugin.DOCUMENTATION_GROUP
197                         t.description = 'Aggregates Javadoc API docs for all projects.'
198                         t.destinationDir = project.file("${project.buildDir}/docs/javadoc")
199                         if (JavaVersion.current().isJava8Compatible()) {
200                             t.options.addBooleanOption('Xdoclint:none', true)
201                             t.options.quiet()
202                         }
203                     }
204                 })
205 
206         project.tasks.register(AGGREGATE_JAVADOCS_JAR_TASK_NAME, Jar,
207                 new Action<Jar>() {
208                     @Override
209                     void execute(Jar t) {
210                         t.dependsOn aggregateJavadocs
211                         t.enabled = false
212                         t.group = JavaBasePlugin.DOCUMENTATION_GROUP
213                         t.description = 'An archive of the aggregate Javadoc API docs'
214                         t.archiveClassifier.set('javadoc')
215                     }
216                 })
217 
218         aggregateJavadocs
219     }

221     @CompileStatic
222     private void createAggregateGroovydocsTask(Project project, TaskProvider<Javadoc> aggregateJavadoc) {
223         TaskProvider<Groovydoc> aggregateGroovydocs = project.tasks.register(AGGREGATE_GROOVYDOCS_TASK_NAME, Groovydoc,
224                 new Action<Groovydoc>() {
225                     @Override
226                     void execute(Groovydoc t) {
227                         t.enabled = false
228                         t.group = JavaBasePlugin.DOCUMENTATION_GROUP
229                         t.description = 'Aggregates Groovy API docs for all projects.'
230                         t.destinationDir = project.file("${project.buildDir}/docs/groovydoc")
231                         t.classpath = aggregateJavadoc.get().classpath
232                     }
233                 })
234 
235         project.tasks.register(AGGREGATE_GROOVYDOCS_JAR_TASK_NAME, Jar,
236                 new Action<Jar>() {
237                     @Override
238                     void execute(Jar t) {
239                         t.dependsOn aggregateGroovydocs
240                         t.enabled = false
241                         t.group = JavaBasePlugin.DOCUMENTATION_GROUP
242                         t.description = 'An archive of the aggregate Groovy API docs'
243                         t.archiveClassifier.set('groovydoc')
244                     }
245                 })
246     }

248     @CompileStatic
249     private void createAggregateApidocTask(Project project) {
250         project.tasks.register(AGGREGATE_APIDOCS_TASK_NAME, DefaultTask, new Action<DefaultTask>() {
251             @Override
252             void execute(DefaultTask t) {
253                 t.enabled = false
254                 t.group = JavaBasePlugin.DOCUMENTATION_GROUP
255                 t.description = 'Aggregates all API docs for all projects.'
256             }
257         })
258     }
