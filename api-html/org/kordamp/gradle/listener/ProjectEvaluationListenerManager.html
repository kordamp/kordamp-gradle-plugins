018 package org.kordamp.gradle.listener
019 
020 import groovy.transform.CompileStatic
021 import org.gradle.BuildAdapter
022 import org.gradle.BuildResult
023 import org.gradle.api.Action
024 import org.gradle.api.Project
025 import org.gradle.api.ProjectEvaluationListener
026 import org.gradle.api.ProjectState
027 import org.gradle.api.execution.TaskExecutionGraph
028 import org.gradle.api.invocation.Gradle
029 import org.slf4j.Logger
030 import org.slf4j.LoggerFactory
031 
032 import java.util.concurrent.ConcurrentHashMap
033 import java.util.concurrent.CopyOnWriteArrayList
034 
035 import static org.kordamp.gradle.util.AnnotationUtils.sortByDependencies
036 
037 /**
038  * @author Andres Almiray
039  * @since 0.37.0
040  */
041 @CompileStatic
042 class ProjectEvaluationListenerManager {
043     private static final Logger LOG = LoggerFactory.getLogger(Project)
044 
045     private final Map<Project, List<ProjectEvaluatedListener>> projectEvaluatedListeners = new ConcurrentHashMap<>()
046     private final List<AllProjectsEvaluatedListener> allProjectsEvaluatedListeners = new CopyOnWriteArrayList<>()
047     private final List<TaskGraphReadyListener> taskGraphReadyListeners = new CopyOnWriteArrayList<>()
048 
049     private static Map<Gradle, ProjectEvaluationListenerManager> instances = [:]
050 
051     static void register(Gradle gradle) {
052         if (!instances.get(gradle)) {
053             ProjectEvaluationListenerManager instance = new ProjectEvaluationListenerManager()
054             instances.put(gradle, instance)
055             gradle.addListener(instance.listener)
056             gradle.taskGraph.whenReady(new Action<TaskExecutionGraph>() {
057                 @Override
058                 void execute(TaskExecutionGraph graph) {
059                     instance.fireTaskGraphReadyListener(gradle.rootProject, graph)
060                 }
061             })
062         }
063     }
064 
065     static void addProjectEvaluatedListener(Project project, ProjectEvaluatedListener listener) {
066         if (listener) {
067             List<ProjectEvaluatedListener> listeners = instances.get(project.rootProject.gradle).projectEvaluatedListeners
068                 .computeIfAbsent(project, { k -> new CopyOnWriteArrayList<>() })
069             if (!listeners.contains(listener)) {
070                 listeners.add(listener)
071             }
072         }
073     }
074 
075     static void addAllProjectsEvaluatedListener(Project rootProject, AllProjectsEvaluatedListener listener) {
076         if (listener && !instances.get(rootProject.gradle).allProjectsEvaluatedListeners.contains(listener)) {
077             instances.get(rootProject.gradle).allProjectsEvaluatedListeners.add(listener)
078         }
079     }
080 
081     static void addTaskGraphReadyListener(Project rootProject, TaskGraphReadyListener listener) {
082         if (listener && !instances.get(rootProject.gradle).taskGraphReadyListeners.contains(listener)) {
083             instances.get(rootProject.gradle).taskGraphReadyListeners.add(listener)
084         }
085     }
086 
087     private Listener listener = new Listener()
088 
089     private void fireTaskGraphReadyListener(Project rootProject, TaskExecutionGraph graph) {
090         Map<String, TaskGraphReadyListener> sortedListeners = sortByDependencies('task-graph', taskGraphReadyListeners, '', 'listener')
091         for (TaskGraphReadyListener listener : sortedListeners.values()) {
092             LOG.debug('[task-graph] Invoking listener {}', listener)
093             listener.taskGraphReady(rootProject, graph)
094         }
095     }
096 
097     private class Listener extends BuildAdapter implements ProjectEvaluationListener {
098         @Override
099         void beforeEvaluate(Project project) {
100             // noop
101         }
102 
103         @Override
104         void afterEvaluate(Project project, ProjectState projectState) {
105             List<ProjectEvaluatedListener> listeners = projectEvaluatedListeners.get(project)
106             if (!listeners) return
107             Map<String, ProjectEvaluatedListener> sortedListeners = sortByDependencies('project-evaluated:' + project.name, listeners, '', 'listener')
108             for (ProjectEvaluatedListener listener : sortedListeners.values()) {
109                 LOG.debug('[project-evaluated:{}] Invoking listener {}', project.name, listener)
110                 listener.projectEvaluated(project)
111             }
112         }
113 
114         @Override
115         void projectsEvaluated(Gradle gradle) {
116             Map<String, AllProjectsEvaluatedListener> sortedListeners = sortByDependencies('all-projects-evaluated', allProjectsEvaluatedListeners, '', 'listener')
117             for (AllProjectsEvaluatedListener listener : sortedListeners.values()) {
118                 LOG.debug('[all-projects-evaluated] Invoking listener {}', listener)
119                 listener.allProjectsEvaluated(gradle.rootProject)
120             }
121         }
122 
123         @Override
124         void buildFinished(BuildResult result) {
125             projectEvaluatedListeners.clear()
126             allProjectsEvaluatedListeners.clear()
127             taskGraphReadyListeners.clear()
128             instances.remove(result.gradle)
129         }
130     }
131 }
