047 class SourceHtmlPlugin implements Plugin<Project> {
048     private static final String VISITED = SourceHtmlPlugin.class.name.replace('.', '_') + '_VISITED'
049 
050     static final String AGGREGATE_SOURCE_HTML_TASK_NAME = 'aggregateSourceHtml'
051     static final String AGGREGATE_CONVERT_CODE_TO_HTML_TASK_NAME = 'aggregateConvertCodeToHtml'
052     static final String AGGREGATE_GENERATE_SOURCE_HTML_OVERVIEW_TASK_NAME = 'aggregateGenerateSourceHtmlOverview'
053     static final String SOURCE_HTML_TASK_NAME = 'sourceHtml'
054     static final String CONFIGURATION_NAME = 'java2html'
055 
056     Project project
057 
058     void apply(Project project) {
059         this.project = project
060 
061         if (isRootProject(project)) {
062             project.childProjects.values().each {
063                 configureProject(it)
064             }
065         }
066         configureProject(project)
067     }
068 
069     static void applyIfMissing(Project project) {
070         if (!project.plugins.findPlugin(SourceHtmlPlugin)) {
071             project.plugins.apply(SourceHtmlPlugin)
072         }
073     }
074 
075     private void configureProject(Project project) {
076         String visitedPropertyName = VISITED + '_' + project.name
077         if (project.findProperty(visitedPropertyName)) {
078             return
079         }
080         project.ext[visitedPropertyName] = true
081 
082         BasePlugin.applyIfMissing(project)
083 
084         Configuration configuration = project.configurations.create(CONFIGURATION_NAME)
085             .setVisible(false)
086             .setTransitive(true)
087             .setDescription('The Java2HTML library to be used for this project.')
088 
089         configuration.incoming.beforeResolve(new Action<ResolvableDependencies>() {
090             @SuppressWarnings('UnusedMethodParameter')
091             void execute(ResolvableDependencies resolvableDependencies) {
092                 DependencyHandler dependencyHandler = project.dependencies
093                 DependencySet dependencies = configuration.dependencies
094                 dependencies.add(dependencyHandler.create('de.java2html:java2html:5.0'))
095             }
096         })
097         project.afterEvaluate {
098             ProjectConfigurationExtension mergedConfiguration = project.ext.mergedConfiguration
099 
100             if (mergedConfiguration.sourceHtml.enabled) {
101                 project.plugins.withType(JavaBasePlugin) {
102                     if (configureSourceHtmlTask(project, configuration).enabled) {
103                         mergedConfiguration.sourceHtml.projects() << project
104                     } else {
105                         mergedConfiguration.sourceHtml.enabled = false
106                     }
107                 }
108             }
109         }
110 
111         if (isRootProject(project) && !project.childProjects.isEmpty()) {
112             project.gradle.addBuildListener(new BuildAdapter() {
113                 @Override
114                 void projectsEvaluated(Gradle gradle) {
115                     configureAggregateSourceHtmlTask(project, configuration)
116                 }
117             })
118         }
119     }
120 
121     private Task configureSourceHtmlTask(Project project, Configuration configuration) {
122         ProjectConfigurationExtension mergedConfiguration = project.ext.mergedConfiguration
123         if (!mergedConfiguration.sourceHtml.enabled) {
124             return
125         }
126 
127         Task classesTask = project.tasks.findByName('classes')
128         if (!classesTask) {
129             return project.tasks.create(SOURCE_HTML_TASK_NAME, DefaultTask) {
130                 enabled = false
131                 group 'Documentation'
132                 description 'Generates a HTML report of the source code'
133             }
134         }
135 
136         mergedConfiguration.sourceHtml.srcDirs = resolveSrcDirs(project, mergedConfiguration.sourceHtml.conversion.srcDirs)
137 
138         Task convertCodeTask = project.tasks.create('convertCodeToHtml', ConvertCodeTask) {
139             enabled = !mergedConfiguration.sourceHtml.srcDirs.isEmpty()
140             dependsOn classesTask
141             group 'Documentation'
142             description 'Converts source code into HTML'
143             classpath = configuration.asFileTree
144             srcDirs = mergedConfiguration.sourceHtml.srcDirs
145             destDir = mergedConfiguration.sourceHtml.conversion.destDir
146             includes = mergedConfiguration.sourceHtml.conversion.includes
147             outputFormat = mergedConfiguration.sourceHtml.conversion.outputFormat
148             tabs = mergedConfiguration.sourceHtml.conversion.tabs
149             style = mergedConfiguration.sourceHtml.conversion.style
150             showLineNumbers = mergedConfiguration.sourceHtml.conversion.showLineNumbers
151             showFileName = mergedConfiguration.sourceHtml.conversion.showFileName
152             showDefaultTitle = mergedConfiguration.sourceHtml.conversion.showDefaultTitle
153             showTableBorder = mergedConfiguration.sourceHtml.conversion.showTableBorder
154             includeDocumentHeader = mergedConfiguration.sourceHtml.conversion.includeDocumentHeader
155             includeDocumentFooter = mergedConfiguration.sourceHtml.conversion.includeDocumentFooter
156             addLineAnchors = mergedConfiguration.sourceHtml.conversion.addLineAnchors
157             lineAnchorPrefix = mergedConfiguration.sourceHtml.conversion.lineAnchorPrefix
158             horizontalAlignment = mergedConfiguration.sourceHtml.conversion.horizontalAlignment
159             useShortFileName = mergedConfiguration.sourceHtml.conversion.useShortFileName
160             overwrite = mergedConfiguration.sourceHtml.conversion.overwrite
161         }
162 
163         Task generateOverviewTask = project.tasks.create('generateSourceHtmlOverview', GenerateOverviewTask) {
164             enabled = !mergedConfiguration.sourceHtml.srcDirs.isEmpty()
165             dependsOn convertCodeTask
166             group 'Documentation'
167             description 'Generate an overview of converted source code'
168             srcDirs = project.files(mergedConfiguration.sourceHtml.conversion.destDir)
169             destDir = mergedConfiguration.sourceHtml.overview.destDir
170             pattern = mergedConfiguration.sourceHtml.overview.pattern
171             windowTitle = mergedConfiguration.sourceHtml.overview.windowTitle
172             docTitle = mergedConfiguration.sourceHtml.overview.docTitle
173             docDescription = mergedConfiguration.sourceHtml.overview.docDescription
174             icon = mergedConfiguration.sourceHtml.overview.icon
175             stylesheet = mergedConfiguration.sourceHtml.overview.stylesheet
176         }
177 
178         project.tasks.create(SOURCE_HTML_TASK_NAME, Copy) {
179             enabled = !mergedConfiguration.sourceHtml.srcDirs.isEmpty()
180             dependsOn generateOverviewTask
181             group 'Documentation'
182             description 'Generates a HTML report of the source code'
183             destinationDir = project.file("${project.buildDir}/docs/source-html")
184             from convertCodeTask.destDir
185             from generateOverviewTask.destDir
186         }
187     }

189     private void configureAggregateSourceHtmlTask(Project project, Configuration configuration) {
190         ProjectConfigurationExtension mergedConfiguration = project.ext.mergedConfiguration
191 
192         Set<Project> projects = new LinkedHashSet<>()
193         FileCollection srcdirs = project.files()
194 
195         project.childProjects.values()*.mergedConfiguration.sourceHtml.each { SourceHtml e ->
196             if (e.enabled) {
197                 projects.addAll(e.projects())
198                 srcdirs = project.files(srcdirs, e.srcDirs)
199             }
200         }
201 
202         Task convertCodeTask = project.tasks.create(AGGREGATE_CONVERT_CODE_TO_HTML_TASK_NAME, ConvertCodeTask) {
203             dependsOn projects.sourceHtml
204             group 'Documentation'
205             description 'Converts source code into HTML'
206             classpath = configuration.asFileTree
207             srcDirs = srcdirs
208             destDir = mergedConfiguration.sourceHtml.conversion.destDir
209             includes = mergedConfiguration.sourceHtml.conversion.includes
210             outputFormat = mergedConfiguration.sourceHtml.conversion.outputFormat
211             tabs = mergedConfiguration.sourceHtml.conversion.tabs
212             style = mergedConfiguration.sourceHtml.conversion.style
213             showLineNumbers = mergedConfiguration.sourceHtml.conversion.showLineNumbers
214             showFileName = mergedConfiguration.sourceHtml.conversion.showFileName
215             showDefaultTitle = mergedConfiguration.sourceHtml.conversion.showDefaultTitle
216             showTableBorder = mergedConfiguration.sourceHtml.conversion.showTableBorder
217             includeDocumentHeader = mergedConfiguration.sourceHtml.conversion.includeDocumentHeader
218             includeDocumentFooter = mergedConfiguration.sourceHtml.conversion.includeDocumentFooter
219             addLineAnchors = mergedConfiguration.sourceHtml.conversion.addLineAnchors
220             lineAnchorPrefix = mergedConfiguration.sourceHtml.conversion.lineAnchorPrefix
221             horizontalAlignment = mergedConfiguration.sourceHtml.conversion.horizontalAlignment
222             useShortFileName = mergedConfiguration.sourceHtml.conversion.useShortFileName
223             overwrite = mergedConfiguration.sourceHtml.conversion.overwrite
224         }
225 
226         Task generateOverviewTask = project.tasks.create(AGGREGATE_GENERATE_SOURCE_HTML_OVERVIEW_TASK_NAME, GenerateOverviewTask) {
227             dependsOn convertCodeTask
228             group 'Documentation'
229             description 'Generate an overview of converted source code'
230             srcDirs = project.files(mergedConfiguration.sourceHtml.conversion.destDir)
231             destDir = mergedConfiguration.sourceHtml.overview.destDir
232             pattern = mergedConfiguration.sourceHtml.overview.pattern
233             windowTitle = mergedConfiguration.sourceHtml.overview.windowTitle
234             docTitle = mergedConfiguration.sourceHtml.overview.docTitle
235             docDescription = mergedConfiguration.sourceHtml.overview.docDescription
236             icon = mergedConfiguration.sourceHtml.overview.icon
237             stylesheet = mergedConfiguration.sourceHtml.overview.stylesheet
238         }
239 
240         project.tasks.create(AGGREGATE_SOURCE_HTML_TASK_NAME, Copy) {
241             dependsOn generateOverviewTask
242             group 'Documentation'
243             description 'Generates a HTML report of the source code'
244             destinationDir = project.file("${project.buildDir}/docs/source-html")
245             from convertCodeTask.destDir
246             from generateOverviewTask.destDir
247         }
248     }

250     private boolean hasJavaPlugin(Project project) {
251         project.plugins.hasPlugin(JavaBasePlugin)
252     }
253 
254     private boolean hasGroovyPlugin(Project project) {
255         project.plugins.hasPlugin(GroovyBasePlugin)
256     }
257 
258     private FileCollection resolveSrcDirs(Project project, FileCollection files) {
259         try {
260             if (project.sourceSets.main) {
261                 if (hasGroovyPlugin(project)) {
262                     return project.files(project.files(files,
263                         project.sourceSets.main.groovy.srcDirs,
264                         project.sourceSets.main.java.srcDirs).files.findAll { file ->
265                         file.exists()
266                     })
267                 } else if (hasJavaPlugin(project)) {
268                     return project.files(
269                         project.files(files,
270                             project.sourceSets.main.java.srcDirs).files.findAll { file ->
271                             file.exists()
272                         })
273                 }
274             }
275         } catch (Exception ignored) {
276             // ignore this project
277             return project.files()
278         }
279 
280         files
281     }
